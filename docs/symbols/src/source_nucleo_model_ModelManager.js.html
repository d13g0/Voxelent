 16 /**
 17  * Creates vxlModel objects and assigns them to a vxlScene. 
 18  * 
 19  * vxlModelManager provides methods for loading remote and local models.
 20  * 
 21  * 
 22  * @class
 23  * @constructor
 24  * @author Diego Cantor
 25  */
 26 function vxlModelManager(){
 27 	this.firstLoadedModel = false; //analyze
 28 	this.toLoad = new Array(); //analyze
 29 	this.models = [];
 30 	vxl.go.notifier.addSource(vxl.events.MODELS_LOADED,this);
 31 };
 32 
 33 
 34 /**
 35  * Uses a JSON/Ajax mechanism to load models from the webserver.
 36  * @param {String} filename name of the file that will be loaded. Voxelent will look for this file inside of the 
 37  * 						folder defined by the configuration variable vxl.def.model.folder
 38  * @param {vxlScene} optional parameter. The scene that will contain an actor for this model.
 39  * @param {Array} attributes to add to the respective actor once the model is being loaded
 40  * @param {Function} callback function once the model is loaded
 41  */  
 42 vxlModelManager.prototype.load = function(filename, scene) {
 43     var manager = this;
 44 	if (manager.isModelLoaded(filename)) return;
 45 	
 46 	vxl.go.console('ModelManager.load: Requesting '+filename+'...');
 47     var request = new XMLHttpRequest();
 48     request.open("GET", filename);
 49     request.onreadystatechange = function() {
 50       if (request.readyState == 4) {
 51 	    if(request.status == 404) {
 52 			throw 'ModelManager.load: '+ filename + ' does not exist';
 53 		 }
 54 		else {
 55 		    
 56 			manager.add(JSON.parse(request.responseText),filename,scene);
 57 		}
 58 	  }
 59     };
 60     request.send();
 61 };
 62 
 63 /**
 64  * Loads a list of models and assigns them to a scene
 65  * @param {Array} list list of files to load 
 66  * @param {vxlScene} scene scene to callback when the models are loaded 
 67  */
 68 vxlModelManager.prototype.loadList = function(list,scene){
 69 	this.toLoad = list.slice(0); 
 70 	vxl.go.console('ModelManager.loadList: models to load ->' + this.toLoad.length);
 71    	for(var i=0;i<this.toLoad.length; i++){
 72 		this.load(this.toLoad[i],scene);
 73    	}
 74 };
 75 
 76 /**
 77  * 
 78  * @param {Object} JSON_OBJECT the JSON object that contains the definition of the model
 79  * @param {String} name name of the model to be created
 80  * @param {vxlScene} scene the scene to be called back when the model is created
 81  */
 82 vxlModelManager.prototype.add = function(JSON_OBJECT,name,scene){
 83 	
 84 	var model = new vxlModel(name, JSON_OBJECT);
 85 	
 86 	
 87 	if (!this.firstLoadedModel){
 88 		scene.bb = model.outline;
 89 		this.firstLoadedModel = true;
 90 	}
 91 		
 92 	model.loaded = true;
 93 	this.models.push(model);
 94 	vxl.go.console('ModelManager: model '+model.name+' created.'); 
 95 	
 96 	if (scene != undefined && scene instanceof vxlScene){
 97 		vxl.go.console('ModelManager: notifying the scene...');
 98 		if (scene.loadingMode == vxl.def.model.loadingMode.LIVE){
 99 			scene.createActor(model);
100 		}
101 		else if (scene.loadingMode == vxl.def.model.loadingMode.LATER){
102 			if(this.allLoaded()){
103 				scene.createActors(this.models);
104 			}
105 		}
106 		else if (scene.loadingMode == vxl.def.model.loadingMode.DETACHED){
107 			//do nothing
108 		}
109 	}
110 	
111 	if(this.allLoaded()){ 
112 		vxl.go.notifier.fire(vxl.events.MODELS_LOADED);
113 	}
114  };

116 /**
117  * It will delete all of the loaded models
118  */  
119 vxlModelManager.prototype.reset = function(){
120 	this.firstLoadedModel = false;
121 	for (var i=0; i <this.models.length; i++){
122 		this.models[i] = null;
123 	}
124 	
125 	this.models        = [];
126 	this.toLoad        = [];
127 };
128 
129 /**
130  * Checks if a model has been loaded yet
131  * @param {String} name the name of the model to check
132  */
133 vxlModelManager.prototype.isModelLoaded = function(name){
134 	for(var i=0;i<this.models.length;i++){
135 		if (this.models[i].name == name) return true;
136 	}
137 	return false;
138 };
139 
140 /**
141  * Returns true if all the models in the list passed to the method loadList.
142  */
143 vxlModelManager.prototype.allLoaded = function(){
144 	return (this.models.length == this.toLoad.length); //TODO: Verify one by one
145 };

148 /**
149  * Returns the model if it has been loaded by this model manager, null otherwise.
150  * @param {String} name the name of the model to retrieve
151  */
152 vxlModelManager.prototype.getModelByName = function(name){
153  	for(var i=0, max = this.models.length; i<max; i+=1){
154 		if (this.models[i].name == name) return this.models[i];
155 	}
156 	return null;
157 };
