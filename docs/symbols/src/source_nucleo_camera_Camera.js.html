 22 /**
 23  * 
 24  * A vxlCamera object simplifies WebGL programming by providing a simple object interface to the lower level
 25  * matrix manipulations that are required to view a 3D scene.
 26  * 
 27  * When moving and rotating the camera, such matrices are updated and Voxelent's Nucleo will use this information
 28  * in order to draw the scene accordingly to the camera position an orientation.
 29  * 
 30  * A vxlCamera object requires a vxlView to be created. Having said that, a vxlView can be associated with 
 31  * multiple cameras.
 32  * 
 33  * @class
 34  * @constructor Creates a vxlCamera. 
 35  * @param {vxlView} vw
 36  * @param {Object} t the type of camera 
 37  * @author Diego Cantor
 38  */
 39 function vxlCamera(vw,t) {
 40 
 41 	this.view 		= vw;
 42     this.matrix 	= new vxlMatrix4x4();
 43     this.up 		= vxl.vec3.create([0, 1, 0]);
 44 	this.right 		= vxl.vec3.create([1, 0, 0]);
 45 	this.normal 	= vxl.vec3.create([0, 0, 0]);
 46     this.position 	= vxl.vec3.create([0, 0, 1]);
 47     this.focus		= vxl.vec3.create([0, 0, 0]);
 48     this.azimuth 	= 0;
 49     this.elevation 	= 0;
 50 	this.steps		= 0;
 51     this.home 		= vxl.vec3.create([0,0,0]);
 52     this.id         = 0;
 53     this.FOV        = 30;
 54     this.Z_NEAR     = 0.1;    
 55     this.Z_FAR      = 10000;     
 56     
 57 	
 58    
 59 	
 60 	if (t != undefined && t !=null){
 61         this.type = t;
 62     }
 63     else {
 64         this.type = vxl.def.camera.type.ORBITING;
 65     }
 66     
 67 
 68 	this.distance = 0;
 69 	this.debug = false;
 70 
 71 };

 74 /**
 75  * Establishes the type of camera
 76  * @param {vxl.def.camera.type} t
 77  */
 78 vxlCamera.prototype.setType = function(t){
 79     if (t != vxl.def.camera.type.ORBITING && t != vxl.def.camera.type.TRACKING) {
 80         alert('Wrong Camera Type!. Setting Orbitting type by default');
 81         this.type = vxl.def.camera.type.ORBITING;
 82     }
 83     else{
 84         this.type = t;
 85     }
 86 };

 88 /**
 89  *	Initializes the camera 
 90  */
 91 vxlCamera.prototype.init = function() {
 92 	var c = this;
 93 	this.goHome([0,0,1]);
 94 	this.setFocus([0,0,0]);
 95 	vxl.mat4.identity(this.matrix);
 96 };

 98 /**
 99  * Sends the camera home. Wherever that home is.
100  * @param {Array} h home
101  */
102 vxlCamera.prototype.goHome = function(h){
103     if (h != null){
104         this.home = vxl.vec3.create(h);
105     }
106     this.setPosition(h);
107     this.setAzimuth(0);
108     this.setElevation(0);
109     this.steps = 0;
110 };

180 vxlCamera.prototype.dolly = function(value) {
181     var c = this;
182     
183     var p =  vxl.vec3.create();
184     var n =  vxl.vec3.create();
185     
186     p = c.position;
187     
188     var step = value - c.steps;
189     
190     vxl.vec3.normalize(c.normal,n);
191     
192     var newPosition = vxl.vec3.create();
193     
194     if(c.type == vxl.def.camera.type.TRACKING){
195         newPosition.x = p.x - step*n.x;
196         newPosition.y = p.y - step*n.y;
197         newPosition.z = p.z - step*n.z;
198     }
199     else{
200         newPosition.x = p.x;
201         newPosition.y = p.y;
202         newPosition.z = p.z - step; 
203     }
204     
205     c.setPosition([newPosition.x, newPosition.y, newPosition.z]); //@TODO: fix this syntax ambivalence
206     c.steps = value;
207 };

217 /**
218  * Changes the azimuth of the camera
219  * @param {Number} az the relative increment in degrees
220  */
221 vxlCamera.prototype.changeAzimuth = function(az){
222     var c = this;
223     c.azimuth +=az;
224     
225     if (c.azimuth > 360 || c.azimuth <-360) {
226 		c.azimuth = c.azimuth % 360;
227 	}
228     c.update();
229 };

255 /**
256  * Updates the x,y and z axis of the camera according to the current camera matrix.
257  * This is useful when one needs to know the values of the axis and operate with them directly.
258  * Such is the case for zooming, where you need to know what is the normal axis to move
259  * along it for dollying or zooming.
260  */
261 vxlCamera.prototype.computeAxis = function(){
262 	var m       = this.matrix;
263     this.right  = vxl.mat4.multVec3(m, [1, 0, 0]);
264     this.up     = vxl.mat4.multVec3(m, [0, 1, 0]);
265     this.normal = vxl.mat4.multVec3(m, [0, 0, 1]);
266 };
267 
268 /**
269  * This method updates the current camera matrix upon changes in location, 
270  * and rotation (azimuth, elevation)
271  */
272 vxlCamera.prototype.update = function(){
273 	vxl.mat4.identity(this.matrix);
274 	
275 	this.computeAxis();
276     
277     if (this.type == vxl.def.camera.type.TRACKING){
278         vxl.mat4.translate(this.matrix, this.position);
279         vxl.mat4.rotateY(this.matrix, this.azimuth * Math.PI/180);
280         vxl.mat4.rotateX(this.matrix, this.elevation * Math.PI/180);
281     }
282     else {
283         var trxLook = new vxlMatrix4x4();
284         vxl.mat4.translate(this.matrix, this.focus);
285         vxl.mat4.rotateY(this.matrix, this.azimuth * Math.PI/180);
286         vxl.mat4.rotateX(this.matrix, this.elevation * Math.PI/180);
287         vxl.mat4.translate(this.matrix, vxl.vec3.negate(this.focus));
288         vxl.mat4.translate(this.matrix,this.position);
289         //mat4.lookAt(this.position, this.focus, this.up, trxLook);
290         //mat4.inverse(trxLook);
291         //mat4.multiply(this.matrix,trxLook);
292     }
293 
294     this.computeAxis();
295     
296     /**
297     * We only update the position if we have a tracking camera.
298     * For an orbiting camera we do not update the position. If
299     * you don't believe me, go ahead and comment the if clause...
300     * Why do you think we do not update the position?
301     */
302     if(this.type == vxl.def.camera.type.TRACKING){
303         vxl.mat4.multVec4(this.matrix, new vxlVector4(0, 0, 0, 1), this.position);
304     }
305     
306     
307     
308     this.debugMessage('------------- update -------------');
309     this.debugMessage(' right: ' + this.right.toString(2)+', up: ' + this.up.toString(2)+', normal: ' + this.normal.toString(2));
310     this.debugMessage('   pos: ' + this.position.toString(2));
311     this.debugMessage('   azimuth: ' + this.azimuth +', elevation: '+ this.elevation);
312 };

314 /**
315  * Inverts the camera mattrix to obtain the correspondent Model-View Transform
316  * @returns {vxlMatrix4x4} m the Model-View Transform
317  */
318 vxlCamera.prototype.getViewTransform = function(){
319     var m = new vxlMatrix4x4();
320     vxl.mat4.set(this.matrix, m);
321     vxl.mat4.inverse(m);
322     return m;
323 };

334 /**
335  * This method sets the camera to a distance such that the area covered by the bounding box (parameter)
336  * is viewed.
337  * @param {vxlBoundingBox} bb the bounding box
338  */
339 vxlCamera.prototype.shot = function(bb){
340 	var maxDim = Math.max(bb[3] - bb[0], bb[4] - bb[1]);
341 	var centre = this.view.scene.centre;
342 	
343 	if(maxDim != 0) {
344 		var distance = 1.5 * maxDim / (Math.tan(this.FOV * Math.PI / 180));
345 		this.setPosition([centre[0], centre[1], centre[2]+ distance]);
346 	}
347 	
348 	this.setFocus(centre);
349 };

352  * The camera moves to a position where all the actors in the scene are viewed. The actors
353  * are seen in full within their surrounding environment.
354  * 
355  * A long shot uses the global bounding box of the view's scene
356  */
357 vxlCamera.prototype.longShot = function() {
358 	this.shot(this.view.scene.bb);
359 };
