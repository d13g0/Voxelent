 22 /**
 23  * 
 24  * A vxlCamera object simplifies WebGL programming by providing a simple object interface to the lower level
 25  * matrix manipulations that are required to view a 3D scene.
 26  * 
 27  * When moving and rotating the camera, such matrices are updated and Voxelent's Nucleo will use this information
 28  * in order to draw the scene accordingly to the camera position an orientation.
 29  * 
 30  * A vxlCamera object requires a vxlView to be created. Having said that, a vxlView can be associated with 
 31  * multiple cameras.
 32  * 
 33  * @class
 34  * @constructor Creates a vxlCamera. 
 35  * @param {vxlView} vw
 36  * @param {Object} t the type of camera 
 37  * @author Diego Cantor
 38  */
 39 function vxlCamera(vw,t) {
 40 
 41     this.id             = 0;
 42     this.FOV            = vxl.def.camera.fov;
 43     this.Z_NEAR         = vxl.def.camera.near;    
 44     this.Z_FAR          = vxl.def.camera.far;
 45 	this.view 		    = vw;
 46     this.matrix 	    = mat4.identity();
 47     this.right 		    = vec3.createFrom(1, 0, 0);
 48 	this.up             = vec3.createFrom(0, 1, 0);
 49 	this.normal         = vec3.createFrom(0, 0, 1);	
 50 	this.position       = vec3.createFrom(0, 0, 1);
 51 	this.focalPoint     = vec3.createFrom(0, 0, 0);
 52 	this.cRot           = vec3.createFrom(0, 0, 0);
 53     this.azimuth 	    = 0;
 54     this.elevation 	    = 0;
 55     this.relAzimuth     = 0;
 56     this.relElevation   = 0;
 57 	this.dstep  	    = 0; //dollying step
 58     this.m              = mat4.identity();
 59     this.distance       = 1;
 60     this.debug          = false;
 61     this.states         = [];
 62 
 63 	if (t != undefined && t !=null){
 64         this.type = t;
 65     }
 66     else {
 67         this.type = vxl.def.camera.type.ORBITING;
 68     }
 69 };
 70 
 71 /**
 72  * Establishes the type of camera
 73  * @param {vxl.def.camera.type} t the type of camera
 74  * @see {vxl.def.camera}
 75  */
 76 vxlCamera.prototype.setType = function(t){
 77     if (t != vxl.def.camera.type.ORBITING && t != vxl.def.camera.type.TRACKING) {
 78         alert('Wrong Camera Type!. Setting Orbitting type by default');
 79         this.type = vxl.def.camera.type.ORBITING;
 80     }
 81     else{
 82         this.type = t;
 83     }
 84 };

 86 /**
 87  * Sets the camera position in the scene
 88  * This method has three parameters x,y,z which represent the coordinates for 
 89  * the camera's position.
 90 
 91  * @param {Number, Array} x the x-coordinate. x can also be an Array [a,b,c] in this case the y and z parameters are discarded.
 92  * @param {Number} y the y-coordinate
 93  * @param {Number} z the z-coordinate
 94  */
 95 vxlCamera.prototype.setPosition = function(x,y,z) {
 96     this.position = vxl.util.createVec3(x,y,z);
 97     this._updateDistance();
 98     
 99     var m = this.matrix;
100     m[12] = this.position[0];
101     m[13] = this.position[1];
102     m[14] = this.position[2];
103     m[15] = 1;
104     this._updateCentreRotation();
105 };

209 /**
210  * Changes the azimuth and elevation with respect to the current camera axes
211  * @param {Number} azimuth the relative azimuth
212  * @param {Number} elevation the relative elevation
213  */
214 vxlCamera.prototype.rotate = function(azimuth,elevation){
215     this.relElevation = this._getAngle(elevation);
216     this.relAzimuth   = this._getAngle(azimuth);
217     
218     this.elevation += this.relElevation;
219     this.azimuth += this.relAzimuth;
220     
221     this._updateMatrix();
222 };
223 
224 /**
225  * Performs the dollying operation in the direction indicated by the camera normal axis.
226  * The dollying mechanism offered by a camera makes sure that the camera moves fast
227  * towards the object when the distance is large and slow when it is very close to the object.
228  * For that effect, every time that the new position  (after dollying) is calculated, the field dstep is computed.
229  * 
230  * @param {Number} value the dollying value 
231  * @see {vxlCamera#_updateDistance}
232  */
233 vxlCamera.prototype.dolly = function(value) {
234     this._updateAxes();
235 	var    n =  this.normal; 
236     var pos = vec3.create(this.position);
237     var step = value*this.dstep;
238     pos[0] += step*n[0];
239     pos[1] += step*n[1];
240     pos[2] += step*n[2];
241     this.setPosition(pos);
242 };

245  * Translates the camera side-to-side and up-and-down
246  * @param {Number} dx the horizontal displacement
247  * @param {Number} dy the vertical displacement
248  */
249 vxlCamera.prototype.pan = function(tx, ty) {
250     this._updateAxes();
251     if (this.type == vxl.def.camera.type.ORBITING){
252 	   mat4.translate(this.matrix,  vec3.negate(this.position, vec3.create()));
253 	   vec3.add(this.position, [tx,-ty,0]);
254 	   mat4.translate(this.matrix, this.position); 
255     }
256     //@TODO: Implement for tracking camera  
257 };

291 /**
292  *@param {String} actorName. The name of the actor that the camera will focus on 
293  */
294 vxlCamera.prototype.focusOn = function(actorName){
295 	var actor = this.view.scene.getActorByName(actorName);
296 	if (actor == undefined){
297 		throw 'The actor '+actorName+' does not exist'
298 	}
299 	this.shot(actor.bb);	
300 	
301 }

303 /**
304  * This method sets the camera to a distance such that the area covered by the bounding box (parameter)
305  * is viewed.
306  * @param {vxlBoundingBox} bb the bounding box
307  */
308 vxlCamera.prototype.shot = function(bb){
309 	var maxDim = Math.max(bb[3] - bb[0], bb[4] - bb[1]);
310 	
311 	cc = [0,0,0];
312 
313 	cc[0] = (bb[3] + bb[0]) /2;
314 	cc[1] = (bb[4] + bb[1]) /2;
315 	cc[2] = (bb[5] + bb[2]) /2;
316 		
317 	cc[0] = Math.round(cc[0]*1000)/1000;
318 	cc[1] = Math.round(cc[1]*1000)/1000;
319 	cc[2] = Math.round(cc[2]*1000)/1000;
320 	
321 	if(maxDim != 0) {
322 		var d = 1.5 * maxDim / (Math.tan(this.FOV * Math.PI / 180));
323 		this.setPosition([cc[0], cc[1], cc[2]+ d]);
324 	}
325 	
326 	this.setFocalPoint(cc);
327 };

332  * 
333  * A long shot uses the global bounding box of the view's scene
334  */
335 vxlCamera.prototype.longShot = function() {

348 /**
349  * Retrieves the last saved camera state from the memento (vxlCameraState)
350  * @see vxlCameraState
351  */
352 vxlCamera.prototype.retrieve = function(name) {
353 	for(var i=0;i<this.states.length;i+=1){
354 	    if (this.states[i].name == name){
355 	        this.states[i].retrieve();
356 	        return;
357 	    }
358 	}
359     throw 'vxlCamera.retrieve: state '+name+' not found'
360 };

382 /**
383  * Sets the rotation matrix according to the azimuth and elevation properties of this camera
384  * @private
385  */
386 vxlCamera.prototype._setInitialMatrix = function(){
387     this.matrix = mat4.identity();
388     
389     mat4.translate(this.matrix, this.focalPoint);
390     mat4.rotateY  (this.matrix, this.azimuth   * Math.PI / 180);
391     mat4.rotateX  (this.matrix, this.elevation * Math.PI / 180);
392     mat4.translate(this.matrix, vec3.negate(this.focalPoint, vec3.create()));
393     mat4.translate(this.matrix, this.position);
394 };
395 
396 /**
397  * This method updates the current camera matrix upon changes in location, 
398  * and rotation (azimuth, elevation)
399  * @private
400  */
401 vxlCamera.prototype._updateMatrix = function(){
402 
403     if (this.type ==  vxl.def.camera.type.TRACKING){
404         //mat4.translate(this.matrix, this.position);
405         //mat4.rotateY(this.matrix, this.azimuth * Math.PI/180);
406         //mat4.rotateX(this.matrix, this.elevation * Math.PI/180);
407               
408      }
409      else if (this.type ==  vxl.def.camera.type.ORBITING){
410                
411         //Rotations according to Tojiro
412         var rotY  = quat4.fromAngleAxis(this.relAzimuth * Math.PI/180, [0,1,0]);
413         var rotX  = quat4.fromAngleAxis(this.relElevation * Math.PI/180, [1,0,0]);
414         var rotQ = quat4.multiply(rotY, rotX, quat4.create());
415         var rotMatrix = quat4.toMat4(rotQ);
416     
417         mat4.translate(this.matrix, this.cRot)
418         mat4.multiply(this.matrix, rotMatrix);
419         mat4.translate(this.matrix, vec3.negate(this.cRot, vec3.create()));
420     
421         this.relAzimuth = 0;
422         this.relElevation = 0;
423     } 
424 };
425 
426 /**
427  * Updates the x,y and z axis of the camera according to the current camera matrix.
428  * This is useful when one needs to know the values of the axis and operate with them directly.
429  * Such is the case for zooming, where you need to know what is the normal axis to move
430  * along it for dollying or zooming.
431  * @private
432  */
433 vxlCamera.prototype._updateAxes = function(){
434     var m       = this.matrix;
435     vec3.set(mat4.multiplyVec4(m, [1, 0, 0, 0]), this.right);
436     vec3.set(mat4.multiplyVec4(m, [0, 1, 0, 0]), this.up);
437     vec3.set(mat4.multiplyVec4(m, [0, 0, 1, 0]), this.normal);
438     vec3.set(mat4.multiplyVec4(m, [0, 0, 0, 1]), this.position);
439     
440     vec3.normalize(this.right);
441     vec3.normalize(this.up);
442     vec3.normalize(this.normal);
443     
444     this._updateCentreRotation();
445 };
446 
447 /**
448  * Calculates the distance to the focal point. This method is called internally by the dollying operation
449  * @private
450  */
451 vxlCamera.prototype._updateDistance = function(){
452     this.distance = vec3.dist(this.focalPoint, this.position);
453     //this method also calculates the dolly step (dstep). 
454     //We make sure that there are always 100 setps between 
455     //the camera position and the focus point AT ALL TIMES.
456     this.dstep = this.distance / 100;
457 };

460 /**
461  * Updates the vector that moves the camera to the center of rotation
462  * This method is supposed to be called internally by other methods that 
463  * update the position or the focal point
464  * @private
465  */
466 vxlCamera.prototype._updateCentreRotation = function(){
467     this.cRot = vec3.subtract(this.focalPoint, this.position, vec3.create());
468     var cMat = mat4.inverse(mat4.toRotationMat(this.matrix), mat4.create());
469     mat4.multiplyVec3(cMat, this.cRot);
470 };

473 /**
474  * Calculates the position based on the current distance
475  * @private 
476  */
477 vxlCamera.prototype._updatePosition = function(){
478     var pos = vec3.create();
479     
480     var d = this.distance;
481     var n = this.normal;
482     
483     pos[0] = d*n[0]; //@TODO: We might need the focal point here
484     pos[1] = d*n[1];
485     pos[2] = d*n[2];
486     
487     vec3.set(pos, this.position);
488     
489     var m = this.matrix;
490     m[12] = this.position[0];
491     m[13] = this.position[1];
492     m[14] = this.position[2];
493     m[15] = 1;
494     
495     this._updateCentreRotation();
496 };
