 17 /**
 18  * 
 19  * A vxlCamera object simplifies WebGL programming by providing a simple object interface to the lower level
 20  * matrix manipulations that are required to view a 3D scene.
 21  * 
 22  * When moving and rotating the camera, such matrices are updated and Voxelent's Nucleo will use this information
 23  * in order to draw the scene accordingly to the camera position an orientation.
 24  * 
 25  * A vxlCamera object requires a vxlView to be created. Having said that, a vxlView can be associated with 
 26  * multiple cameras.
 27  * 
 28  * @class Like a real camera it moves around the 3D scene displaying the current point of view in the browser
 29  * @constructor Creates a vxlCamera. 
 30  * @param {vxlView} vw
 31  * @param {Object} t the type of camera 
 32  * @author Diego Cantor
 33  */
 34 function vxlCamera(vw,t) {
 35 
 36     this.id             = 0;    //Used by the camera manager to switch cameras
 37     this.UID            = vxl.util.generateUID(); //unique identification key
 38     this.view           = vw;
 39 
 40     this.FOV            = vxl.def.camera.fov;
 41     this.Z_NEAR         = vxl.def.camera.near;    
 42     this.Z_FAR          = vxl.def.camera.far;
 43     
 44     this._matrix 	      = mat4.identity();
 45     this._right 		  = vec3.createFrom(1, 0, 0);
 46 	this._up              = vec3.createFrom(0, 1, 0);
 47 	this._normal          = vec3.createFrom(0, 0, 1);	
 48 	this._position        = vec3.createFrom(0, 0, 1);
 49 	this._focalPoint      = vec3.createFrom(0, 0, 0);
 50 	this._distanceVector  = vec3.createFrom(0, 0, 0);
 51     
 52     this._azimuth 	    = 0;
 53     this._elevation     = 0;
 54     this._roll          = 0;
 55 	this._relAzimuth    = 0;
 56     this._relElevation  = 0;
 57     this._relRoll       = 0;
 58 	this._dollyingStep  = 0; //dollying step
 59     this._distance      = 1;
 60     
 61     this._following     = undefined;
 62     this._trackingMode  = vxl.def.camera.tracking.DEFAULT;
 63     
 64     this.states         = [];
 65 
 66 	if (t != undefined && t !=null){
 67         this.type = t;
 68     }
 69     else {
 70         this.type = vxl.def.camera.type.ORBITING;
 71     }
 72 };

 81 /**
 82  * Establishes the type of camera
 83  * @param {vxl.def.camera.type} type the type of camera
 84  * @param {vxl.def.camera.tracking} trackingMode if the camera is of tracking type, the tracking mode can be set as 
 85  * an optional parameter here.
 86  * @see {vxl.def.camera}
 87  */
 88 vxlCamera.prototype.setType = function(type, trackingMode){
 89     
 90     if (type != vxl.def.camera.type.ORBITING && type != vxl.def.camera.type.TRACKING) {
 91         console.error('vxlCamera.setType ERROR type'+ type +' unknown. Setting camera to ORBITING type.');
 92         this.type = vxl.def.camera.type.ORBITING;
 93     }
 94     else {
 95         this.type = type
 96     }
 97     
 98     if (this.type  == vxl.def.camera.type.TRACKING && trackingMode != undefined){
 99         this.setTrackingMode(trackingMode);
100     }
101 };

103 /**
104  * Sets the tracking mode of this camera when it follows an actor
105  * <p> to set the tracking mode of the camera <code>myCamera</code> you should make sure that your camera is of tracking type with:
106  *  <code>myCamera.setType(vxl.def.camera.type.TRACKING)</code>.
107  *  For instance:
108  * </p>
109  *  <pre class='prettyprint'>
110  *  var actor = vxl.api.getActor('cone'); //from the current scene
111  *  var camera = vxl.c.camera;
112  *  camera.setType(vxl.def.camera.type.TRACKING);
113  *  camera.setTrackingMode(vxl.def.camera.tracking.ROTATIONAL);
114  *  camera.follow(actor);
115  * </pre>
116  * <p> a shorter way would be:</p>
117  * <pre class='prettyprint'>
118  *  var actor = vxl.api.getActor('cone'); //from the current scene
119  *  var camera = vxl.c.camera;
120  *  camera.setType(vxl.def.camera.type.TRACKING);
121  *  camera.follow(actor, vxl.def.camera.tracking.ROTATIONAL);
122  * </pre>
123  * @see vxlCamera#follow, vxlCamera#setTrackingMode
124  */
125 vxlCamera.prototype.setTrackingMode = function(mode){
126     if (mode == undefined) return;
127     this._trackingMode = mode;
128 };

152 /**
153  * Updates the camera according to the current tracking mode. This method is a callback used by the 
154  * vxlActor class to notify the camera of any actor transformations. This is applicable of course only when
155  * this camera is following the actor passed as parameter here
156  * @param {vxlActor} actor the actor being followed by this camera
157  */
158 vxlCamera.prototype.updateWithActor = function(actor){
159     if (this._following != actor) return; //fail safe
160     
161     switch(this._trackingMode){
162         case vxl.def.camera.tracking.DEFAULT:
163                 break;
164         case vxl.def.camera.tracking.ROTATIONAL:
165                 this.setFocalPoint(actor._position);
166                 break;
167         case vxl.def.camera.tracking.TRANSLATIONAL:
168                 this.translate(actor._translation);        
169                 break;
170     }
171 };
172 /**
173  *<p>Sets the camera position in the scene
174  * This method has three parameters x,y,z which represent the coordinates for 
175  * the camera's position.
176  * </p>
177  * <p>
178  * This method takes into account the current focal point. The camera will look at the
179  * focal point after this operation. If you want to move the camera position and the focal point
180  * simultaneously, then use <code>vxlCamera.translate</code>.
181  * </p>
182  *  
183  *  * @param {Number, Array} x the x-coordinate. x can also be an Array [a,b,c] in this case the y and z parameters are discarded.
184  * @param {Number} y the y-coordinate
185  * @param {Number} z the z-coordinate
186  * @see{vxlCamera#translate}
187  */
188 vxlCamera.prototype.setPosition = function(x,y,z) {
189     this._position = vxl.util.createVec3(x,y,z);
190     
191     var m = this._matrix;
192     m[12] = this._position[0];
193     m[13] = this._position[1];
194     m[14] = this._position[2];
195     m[15] = 1;
196     
197     this._updateDistance();
198     this.setFocalPoint(this._focalPoint);
199 };
200 
201 /**
202  * Position the camera to a given distance from the current focal point
203  * @param {Number} d the distance
204  * @TODO: REVIEW COMMENTED CODE
205  */
206 vxlCamera.prototype.setDistance = function(d) {
207     
208     if(this._distance == d || d <0) {
209         return;
210     }
211 
212     this._distance = d;
213 
214     // Distance should be greater than .0002
215     if(this._distance < 0.0002) {
216         this._distance = 0.0002;
217         console.warn(" vxlCamera.setDistance WARN: Distance is set to minimum (0.0002)");
218     }
219     
220     this._dollyingStep = this._distance / 100;
221     
222     var pos = vec3.create();
223     
224     var d = this._distance;
225     var n = this._normal;
226     var f = this._focalPoint;
227     
228     pos[0] = d*n[0] + f[0];
229     pos[1] = d*n[1] + f[1];
230     pos[2] = d*n[2] + f[2];
231     
232     vec3.set(pos, this._position);
233     
234     var m = this._matrix;
235     m[12] = this._position[0];
236     m[13] = this._position[1];
237     m[14] = this._position[2];
238     m[15] = 1;
239     
240     this._updateDistance();
241 };

295 /**
296  * Changes the azimuth and elevation with respect to the current camera axes
297  * @param {Number} azimuth the relative azimuth
298  * @param {Number} elevation the relative elevation
299  */
300 vxlCamera.prototype.rotate = function(azimuth,elevation,roll){
301     this._relElevation = this._getAngle(elevation);
302     this._relAzimuth   = this._getAngle(azimuth);
303     this._relRoll      = this._getAngle(roll);
304     
305     this._elevation += this._relElevation;
306     this._azimuth   += this._relAzimuth;
307     this._roll      += this._relRoll;
308     
309     this._rotateMatrix();
310     this._updateAxes();
311     this._updateAngles();
312 };

314 /**
315  * Performs the dollying operation in the direction indicated by the camera normal axis.
316  * The dollying mechanism offered by a camera makes sure that the camera moves fast
317  * towards the object when the distance is large and slow when it is very close to the object.
318  * For that effect, every time that the new position  (after dollying) is calculated, the field dstep is computed.
319  * 
320  * @param {Number} value the dollying value 
321  * @see {vxlCamera#_updateDistance}
322  */
323 vxlCamera.prototype.dolly = function(value) {
324     this._updateAxes();
325 	var    n =  this._normal; 
326     var pos = vec3.create(this._position);
327     var step = value*this._dollyingStep;
328     pos[0] += step*n[0];
329     pos[1] += step*n[1];
330     pos[2] += step*n[2];
331     this.setPosition(pos);
332 };

343 /**
344  * Translates the camera by a given vector 
345  * 
346  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
347  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
348  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
349  */
350 vxlCamera.prototype.translate = function(x,y,z){
351     
352     this._updateAxes();
353     
354     var pos = vec3.create(this._position);
355     
356     vec3.add(pos, vec3.scale(this._right  ,x, vec3.create()));
357     vec3.add(pos, vec3.scale(this._up     ,y, vec3.create()));    
358     vec3.add(pos, vec3.scale(this._normal ,z, vec3.create()));
359     
360     var fp  = vec3.create(this._focalPoint);
361     vec3.add(fp, vec3.scale(this._right  ,x, vec3.create()));
362     vec3.add(fp, vec3.scale(this._up     ,y, vec3.create()));    
363     vec3.add(fp, vec3.scale(this._normal ,z, vec3.create()));
364     
365     
366     this.setPosition(pos);
367     this.setFocalPoint(fp);
368 };

371 /**
372  * Looks at a given point in space (sets the focal point of this camera)
373  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
374  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
375  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
376  */
377 vxlCamera.prototype.setFocalPoint = function(x,y,z) {
378     
379     this._focalPoint = vxl.util.createVec3(x,y,z);
380     
381     this._updateAxes();
382     mat4.inverse(mat4.lookAt(this._position, this._focalPoint, this._up), this._matrix);
383     this._updateAxes();
384     this._updateAngles();
385 };

450 /**
451  * This method sets the camera to a distance such that the area covered by the bounding box (parameter)
452  * is viewed.
453  * @param {vxlBoundingBox} bb the bounding box
454  */
455 vxlCamera.prototype._shot = function(bb){
456     
457     this.setElevation(0);
458     this.setAzimuth(0);
459     
460 	var maxDim = Math.max(bb[3] - bb[0], bb[4] - bb[1]);
461 	
462 	cc = [0,0,0];
463 
464 	cc[0] = (bb[3] + bb[0]) /2;
465 	cc[1] = (bb[4] + bb[1]) /2;
466 	cc[2] = (bb[5] + bb[2]) /2;
467 		
468 	cc[0] = Math.round(cc[0]*1000)/1000;
469 	cc[1] = Math.round(cc[1]*1000)/1000;
470 	cc[2] = Math.round(cc[2]*1000)/1000;
471 	
472 	if(maxDim != 0) {
473 		var d = 1.5 * maxDim / (Math.tan(this.FOV * Math.PI / 180));
474 		this.setPosition([cc[0], cc[1], cc[2]+ d]);
475 	}
476 	
477 	this.setFocalPoint(cc);
478 };

538 /**
539  * Sets the initial camera matrix
540  * @private
541  */
542 vxlCamera.prototype._setInitialMatrix = function(){
543     
544     mat4.identity(this._matrix);
545     
546     var rotX  = quat4.fromAngleAxis(this._elevation * vxl.def.deg2rad, [1,0,0]);
547     var rotY  = quat4.fromAngleAxis(this._azimuth   * vxl.def.deg2rad, [0,1,0]);
548     var rotZ  = quat4.fromAngleAxis(this._roll      * vxl.def.deg2rad, [0,0,1]); 
549     var rotQ = quat4.multiply(rotY, rotX, quat4.create());
550     var rotMatrix = quat4.toMat4(rotQ);
551     
552     mat4.translate(this._matrix, this._focalPoint);
553     mat4.multiply(this._matrix, rotMatrix);
554     var r = vec3.subtract(this._position, this._focalPoint, vec3.create());
555     mat4.translate(this._matrix, r);
556 };
557 
558 /**
559  * This method updates the current camera matrix upon changes in location, 
560  * and rotation (azimuth, elevation)
561  * @private
562  */
563 vxlCamera.prototype._rotateMatrix = function(){
564 
565     var rotX  = quat4.fromAngleAxis(this._relElevation * vxl.def.deg2rad, [1,0,0]); 
566     var rotY  = quat4.fromAngleAxis(this._relAzimuth   * vxl.def.deg2rad, [0,1,0]);
567     var rotZ  = quat4.fromAngleAxis(this._relRoll      * vxl.def.deg2rad, [0,0,1]);
568     var rotQ = quat4.multiply(rotX, rotY, quat4.create());
569     rotQ = quat4.multiply(rotQ, rotZ, quat4.create());
570     var rotMatrix = quat4.toMat4(rotQ);
571     
572     
573     if (this.type ==  vxl.def.camera.type.TRACKING){
574         
575         mat4.multiply(this._matrix, rotMatrix);
576         
577         mat4.multiplyVec3(this._matrix, this._distanceVector);              //updating distance vector and focal point
578         vec3.add(this._position, this._distanceVector, this._focalPoint);
579               
580     }
581     else if (this.type ==  vxl.def.camera.type.ORBITING){
582         mat4.translate(this._matrix, [0,0,-this._distance]);
583         mat4.multiply (this._matrix, rotMatrix);
584         mat4.translate(this._matrix, [0,0,this._distance]);
585     }
586     
587     this._relAzimuth   = 0;
588     this._relElevation = 0;
589     this._relRoll      = 0;
590 };

594 /**
595  * Updates the x,y and z axis of the camera according to the current camera matrix.
596  * This is useful when one needs to know the values of the axis and operate with them directly.
597  * Such is the case for zooming, where you need to know what is the normal axis to move
598  * along it for dollying or zooming.
599  * @private
600  */
601 vxlCamera.prototype._updateAxes = function(){
602    
603     var m       = this._matrix;
604     vec3.set(mat4.multiplyVec4(m, [1, 0, 0, 0]), this._right);
605     vec3.set(mat4.multiplyVec4(m, [0, 1, 0, 0]), this._up);
606     vec3.set(mat4.multiplyVec4(m, [0, 0, 1, 0]), this._normal);
607     vec3.set(mat4.multiplyVec4(m, [0, 0, 0, 1]), this._position);
608     
609     vec3.normalize(this._right);
610     vec3.normalize(this._up);
611     vec3.normalize(this._normal);
612     
613 
614     this._updateDistance();
615 };

618 /**
619  * Updates the vector that moves the camera to the center of rotation. Called internally
620  * when the focal point or the position of the camera are updated
621  * @private
622  */
623 vxlCamera.prototype._updateDistance = function(){
624     this._distanceVector = vec3.subtract(this._focalPoint, this._position, vec3.create());
625     this._distance = vec3.length(this._distanceVector);
626     this._dollyingStep = this._distance / 100;
627 };
628 
629 /**
630  * Calculate the standard angles
631  * @private
632  */
633 vxlCamera.prototype._updateAngles = function(){
634     var p = this._position, f = this._focalPoint;
635        
636     var d = vec3.subtract(f, p, vec3.create());  //distance vector
637     
638     var x = d[0], y =d[1], z =d[2], r = vec3.length(d);
639     
640     
641     var ele = Math.asin(y/r) * vxl.def.rad2deg;
