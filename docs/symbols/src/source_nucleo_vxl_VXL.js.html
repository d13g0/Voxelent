436 util : {
437     /**
438      * Formats Arrays, vec3 and vec4 for display
439      * 
440      * @param {Array, vec3, vec4} arr the object to format
441      * @param {Number} digits the number of decimal figures
442      */
443 	format: function(arr, digits){
444 		var p = Math.pow(10,digits);
445 		if (typeof(arr) == 'object'){
446 			
447 			var result = '['; 
448 			for (var i=0; i < arr.length-1; i+=1){
449 				result  += Math.round(arr[i] * p) / p + ', '; 
450 			}
451 			result += Math.round(arr[arr.length-1] * p) / p  + ']'
452 		}
453 		else if (typeof(arr) == 'number'){
454 			result = '[' + Math.round(arr * p) / p  + ']';
455 		}
456 		return result;
457 	},
458 	
459 	/**
460 	 * Creates a vector from a set of parameters
461 	 * @param {Array, vec3, Number} x it can be an Array, a vec3 or a number
462 	 * @param {Number} y if x is a number, this parameter corresponds to the y-component
463 	 * @param {Number} z if x is a number, this parameter corresponds to the z-component
464 	 */
465 	createVec3: function(x,y,z){
466 	    var vvv = vec3.create();
467 	    if (x instanceof Array){
468             vec3.set(vec3.create(x), vvv)
469         }
470         else if (x instanceof determineMatrixArrayType()){
471             vec3.set(x, vvv)
472         }
473         else{
474             vec3.set(vec3.createFrom(x,y,z), vvv);
475         }
476         return vvv;
477 	},
478 	/**
479      * Creates an array from a set of parameters
480      * @param {Array, vec3, Number} x it can be an Array, a vec3 or a number
481      * @param {Number} y if x is a number, this parameter corresponds to the y-component
482      * @param {Number} z if x is a number, this parameter corresponds to the z-component
483      */
484 	createArr3: function(x,y,z){
485 	    var vvv = []
486         if (x instanceof Array || x instanceof determineMatrixArrayType()){
487             vvv[0] = x[0];
488             vvv[1] = x[1];
489             vvv[2] = x[2];
490         }
491         else{
492             vvv[0] = x;
493             vvv[1] = y;
494             vvv[2] = z;
495         }
496         return vvv;
497 	},

513           }

522         }

525         }
526         else if (path.lastIndexOf('.') > path.lastIndexOf('/')){
527             return path.substring(0, path.lastIndexOf('/')+1)
528         }
529         else

574         }
