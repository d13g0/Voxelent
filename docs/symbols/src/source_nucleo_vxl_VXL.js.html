265 /**
266  * Utility functions
267  * @namespace Voxelent Util methods
268  * 
269  */
270 util : {
271     /**
272      * Formats Arrays, vec3 and vec4 for display
273      * 
274      * @param {Array, vec3, vec4} arr the object to format
275      * @param {Number} digits the number of decimal figures
276      */
277 	format: function(arr, digits){
278 		var p = Math.pow(10,digits);
279 		if (typeof(arr) == 'object'){
280 			
281 			var result = '['; 
282 			for (var i=0; i < arr.length-1; i+=1){
283 				result  += Math.round(arr[i] * p) / p + ', '; 
284 			}
285 			result += Math.round(arr[arr.length-1] * p) / p  + ']'
286 		}
287 		else if (typeof(arr) == 'number'){
288 			result = '[' + Math.round(arr * p) / p  + ']';
289 		}
290 		return result;
291 	},
292 	
293 	/**
294 	 * Creates a vector from a set of parameters
295 	 * @param {Array, vec3, Number} x it can be an Array, a vec3 or a number
296 	 * @param {Number} y if x is a number, this parameter corresponds to the y-component
297 	 * @param {Number} z if x is a number, this parameter corresponds to the z-component
298 	 */
299 	createVec3: function(x,y,z){
300 	    var vvv = vec3.create();
301 	    if (x instanceof Array){
302             vec3.set(vec3.create(x), vvv)
303         }
304         else if (x instanceof determineMatrixArrayType()){
305             vec3.set(x, vvv)
306         }
307         else{
308             vec3.set(vec3.createFrom(x,y,z), vvv);
309         }
310         return vvv;
311 	},

347 Array.prototype.hasObject = (
348   !Array.indexOf ? function (o)
349   {
350     var l = this.length + 1;
351     while (l -= 1)
352     {
353         if (this[l - 1] === o)

355             return true;
356         }
357     }
