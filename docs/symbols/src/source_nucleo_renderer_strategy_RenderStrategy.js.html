 16 /**
 17  * @class
 18  *  A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 19  * 	communicates with the program) from the scene.
 20  * 
 21  *  Any rendering strategy should extend from vxlRenderStrategy or one of its descendants
 22  * @param {vxlRenderer} renderer the renderer this strategy belongs to
 23  */
 24 function vxlRenderStrategy(renderer){
 25 	this.renderer = renderer;
 26 }
 27 
 28 
 29 /**
 30  * @param {vxlScene} scene the scene to allocate memory for
 31  */
 32 vxlRenderStrategy.prototype.allocate = function(scene){
 33 	//DO NOTHING. THE DESCENDANTS WILL.
 34 }
 35 
 36 
 37 /**
 38  * @param {vxlScene} scene the scene to deallocate memory from
 39  */
 40 vxlRenderStrategy.prototype.deallocate = function(scene){
 41 	//DO NOTHING. THE DESCENDANTS WILL.
 42 }
 43 
 44 /**
 45  * @param {vxlScene} scene the scene to render
 46  */
 47 vxlRenderStrategy.prototype.render = function(scene){
 48 	//DO NOTHING. THE DESCENDANTS WILL.
 49 }
