 32 vxlPickerInteractor.prototype.getType = function(){
 33     return "vxlPickerInteractor";
 34 };
 35 
 36 vxlPickerInteractor.prototype.get2DCoords = function(ev){
 37     var x, y, top = 0, left = 0, obj = this.view.canvas;
 38     var rect = obj.getBoundingClientRect();
 39      
 40     // return relative mouse position
 41     x = ev.clientX - rect.left;
 42     y = vxl.c.view.canvas.height - (ev.clientY - rect.top); 
 43                                        //this variable contains the height of the canvas and it updates dynamically
 44                                        //as we resize the browser window.
 45     console.info('x='+x+', y='+y);
 46     return {x:x,y:y};
 47 };
 48 
 49 
 50 /**
 51  *  Reacts to the onmouse up event on the canvas
 52  * @param {Object} ev
 53  */
 54 vxlPickerInteractor.prototype.onMouseUp   = function(ev){
 55     
 56     var rt     = this.view.renderer._renderTarget;
 57     var coords = this.get2DCoords(ev);
 58     var color  = rt.readPixel(coords);
 59     
 60     
 61     var results = vxl.go.picker.query(color);
 62     
 63     if (results == null) return;
 64     
 65     var actor  = this.view.scene.getActorByCellUID(results.uid);
 66     
 67     if (actor != null){
 68         console.info('vxlPickerInteractor: actor ['+actor.name+'] has been picked');
 69        
 70         if (actor.isPickable()){
 71             actor._pickingCallback(actor, results.uid);
 72         }
 73     }
 74 };
