 16 /**
 17  * @private not ready yet...
 18  * @constructor
 19  * @class
 20  * Selects the actors that have a given visualization mode (vxl.def.actor.mode)
 21  * This class helps the vxlBakeEngine
 22  */
 23 function vxlSilo(mode){
 24     this._mode = mode;
 25     this._actors = [];
 26     this.data = {};
 27     this.cached = {};
 28     this.added = {};
 29     this.prepare();
 30 };
 31 
 32 /**
 33  * Populates a silo with the actors from the list that have the type defined for this silo
 34  */
 35 vxlSilo.prototype.populate = function(list){
 36     //this._actors = [];
 37     var N = list.length
 38     for(var i=0; i<N; i+=1){
 39         var actor = list[i];
 40         if (this.cached[actor.UID] == undefined && actor.mode == this._mode){
 41             this._actors.push(actor);
 42             this.cached[actor.UID] = actor;
 43         }
 44     }
 45 };
 46 
 47 /**
 48  * @private
 49  */
 50 vxlSilo.prototype.prepare = function(){
 51 
 52     var data ={
 53         mixin: [],  //vertices + normals +
 54                     // + vertex colors or zeros if not using vertex colors.
 55                     
 56         index : [],  //indices
 57         
 58         matrix : [[],[],[],[]]
 59     };
 60     this.data = data;
 61 };
 62 
 63 vxlSilo.prototype.process = function(){
 64     //this.prepare();
 65     var N = this._actors.length;
 66     for (var i = 0; i < N; i += 1){
 67         var actor = this._actors[i];
 68         if (this.added[actor.UID] == undefined){
 69             this.processActor(this._actors[i]);
 70             this.added[actor.UID] = actor;
 71         }
 72     }  
 73 };
 74 
 75 
 76 vxlSilo.prototype.processActor = function(actor){
 77     
 78     var model = actor.model;
 79     var material = actor.material;
 80     
 81     var mixin = this.data.mixin;
 82     var index = this.data.index;
 83     var matrix = this.data.matrix;
 84     
 85     var num_vertices = model.vertices.length;
 86     var num_indices  = model.indices.length;
 87     
 88     //converting from Float32Array to regular js array to operate
 89     var actor_matrix = Array.prototype.slice.call(actor._matrix);
 90  
 91     
 92     // processing mixin
 93     if (material.colors != undefined){
 94         /*for (var i=0;i<num_vertices; i+=3){
 95             mixin.push(model.vertices[i]);
 96             mixin.push(model.vertices[i+1]);
 97             mixin.push(model.vertices[i+2]);
 98             mixin.push(model.normals[i]);
 99             mixin.push(model.normals[i+1]);
100             mixin.push(model.normals[i+2]);
101             mixin.push(material.colors[i]);
102             mixin.push(material.colors[i+1]);
103             mixin.push(material.colors[i+2]);
104             matrix[0] = matrix[0].concat(actor_matrix.slice(0,4)); //glMatrix IS column-major after all :-P
105             matrix[1] = matrix[1].concat(actor_matrix.slice(4,8));
106             matrix[2] = matrix[2].concat(actor_matrix.slice(8,12));
107             matrix[3] = matrix[3].concat(actor_matrix.slice(12,16));
108         }*/
109     }
110     else{
111         for (var i=0;i<num_vertices; i+=3){
112            
113             mixin = mixin.concat([
114                 model.vertices[i],
115                 model.vertices[i+1],
116                 model.vertices[i+2],
117                 model.normals[i],
118                 model.normals[i+1],
119                 model.normals[i+2],
120                 material.diffuse[0], 
121                 material.diffuse[1],
122                 material.diffuse[2]
123                 ]);
124            
125             matrix[0] = matrix[0].concat(actor_matrix.slice(0,4));
126             matrix[1] = matrix[1].concat(actor_matrix.slice(4,8));
127             matrix[2] = matrix[2].concat(actor_matrix.slice(8,12));
128             matrix[3] = matrix[3].concat(actor_matrix.slice(12,16));
129         }
130     }
131     
132     //processing index
133     var ind = model.indices.slice(0);
134     if (index.length > 0){
135         var offset = index.max()+1;
136         for (var i=0; i<num_indices; i+=1){
137             ind[i] += offset;
138         }
139         index = index.concat(ind);
140     }
141     else{
142         index = ind;
143     }
144     
145     this.data.mixin = mixin;
146     this.data.index = index;
147     this.data.matrix = matrix;
148 
149 };
