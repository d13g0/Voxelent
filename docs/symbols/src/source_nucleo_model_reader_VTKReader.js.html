 23

 25     this.vertices   = [];
 26     this.indices    = [];
 27     this.normals    = [];
 28     this.scalars    = [];
 29     this.colors     = [];
 30 
 31     this.mode = "SOLID";
 32 
 33     this.tags = {
 34         NOWHERE             : 0,    
 35         POINTS              : 1,
 36         LINES               : 2,
 37         POLYGONS            : 3,
 38         POINT_DATA          : 4,
 39         NORMALS             : 5,
 40         CELL_DATA           : 6,
 41         TEXTURE_COORDINATES : 7,
 42         SCALARS             : 8,

 56 /**
 57  * @param {File} file an HTML5 File object
 58  */
 59 vxlVTKReader.prototype.read = function(file){
 60     var vtkReader = this;
 61     
 62     var reader = new FileReader();
 63     reader.onload = function(event){
 64         var contents = event.target.result;
 65         var lines = contents.split(/\r\n|\r|\n/);
 66         vtkReader._parse(lines);  
 67     };
 68     
 69     reader.readAsText(file);
 70 };

212 vxlVTKReader.prototype._weaveBlock = function(blockID, ver, ind, pod, clr){
213 /*    
214 # -----------------------------------------------------------------------
215 # Calculates new index array for the block in question
216 # ver - vertices
217 # ind - indices
218 # seg - segment
219 # -----------------------------------------------------------------------
220 def weaveBlock(blockID, ver,ind, pod,clr):
221     lowerBound  = ARRAY_SIZE*blockID
222     upperBound = ARRAY_SIZE*(blockID+1)
223     if upperBound > len(ind):
224         upperBound = len(ind)
225     newindex = dict()
226     vtxBlock = []   # new vertex array
227     idxBlock = []   # new index array
228     pdxBlock = []   # new scalar array
229     clrBlock = []   # new color array
230     
231     hasPointData = len(pod)>0
232     hasColorData = len(clr)>0

234     # Set of indices to be processed
235     aux = ind[lowerBound:upperBound]
236     taux = len(aux)
237     nidx = -1
238     #item = 1
239     
240     print('Weaving block #' + str(blockID+1) + '  ['+str(lowerBound)+','+str(upperBound)+']')

243     for i,oidx in enumerate(aux):
244         sys.stdout.write('Processing index %d out of %d \r' % (i,taux))

246         if oidx not in newindex.keys():
247             nidx = nidx + 1

249             idxBlock.append(nidx)
250             # save in the map for posterior searches

265                 clrBlock.append(clr[index+2])
266         else:
267             # if the index was mapped then use it in the new index array
268             idxBlock.append(newindex[oidx])
269     print('')
270     del aux
271     return vtxBlock, idxBlock, pdxBlock, clrBlock
