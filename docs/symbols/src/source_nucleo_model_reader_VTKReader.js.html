 17 /**
 18  * Reads a VTK file and creates the respective JSON(s) file(s).
 19  * @class Reads VTK files (ASCII)  and creates JSON objects that can be used as models
 20  * @constructor 
 21  */
 22 function vxlVTKReader(scene){
 23     this.scene      = scene; //the scene that will be updated
 24     vxl.go.notifier.publish(vxl.events.READER_DONE, this);
 25 };

 27 /**
 28  * Verifies if the HTML5 API is available. 
 29  */
 30 vxlVTKReader.prototype.isSupported = function(){
 31     return (window.File && window.FileReader && window.FileList && window.Blob);
 32 };

 35 /**
 36  * @param {File} file an HTML5 File object
 37  */
 38 vxlVTKReader.prototype.readFile = function(file){
 39     var vtkReader = this;
 40     var filename = file.name;
 41     
 42     if (filename.length - 4 !== filename.indexOf('.vtk')){
 43         throw 'vxlVTKReader.read: the file '+file+' is not a VTK file';
 44     }
 45     
 46     modelname = filename.replace(/\.[^/.]+$/, "");
 47     var freader = new FileReader();
 48      
 49     freader.onload = function(event){
 50         document.body.style.cursor = 'default';
 51         var contents = event.target.result.trim();
 52         var lines = contents.split(/\r\n|\r|\n/);
 53         vtkReader._parse(modelname,lines);  
 54         vxl.go.notifier.fire(vxl.events.READER_DONE, vtkReader);
 55     };
 56     
 57     document.body.style.cursor = 'wait';
 58     freader.readAsText(file);
 59 };

 62 /**
 63  * If the VTK file exists in memory as a String (text) then we can
 64  * use this method to parse it. 
 65  * 
 66  * @param {String} name the name of the object
 67  * @param {String} text the retrieved object contents
 68  */
 69 vxlVTKReader.prototype.readText = function(name, text){
 70     document.body.style.cursor = 'wait'; 
 71     var lines = text.split(/\r\n|\r|\n/);
 72     this._parse(name,lines);  
 73     vxl.go.notifier.fire(vxl.events.READER_DONE, this);
 74     document.body.style.cursor = 'default';
 75 };

 77 /**
 78  * Reads the file line by line and populates the respective arrays
 79  */
 80 vxlVTKReader.prototype._parse = function(name,lines){
 81     
 82    var ARRAY_SIZE = 65536*3;
 83    var outputfile = '';
 84    var numBlocks = 0;
 85    var location = 'NOWHERE';
 86    var linenumber = 0;
 87    
 88    
 89    var vertices   = [];
 90    var indices    = [];
 91    var normals    = [];
 92    var scalars    = [];
 93    var colors     = [];
 94    var mode = "SOLID";
 95    
 96    var tags = {
 97         NOWHERE             : 0,    
 98         POINTS              : 1,
 99         LINES               : 2,
100         POLYGONS            : 3,
101         POINT_DATA          : 4,
102         NORMALS             : 5,
103         CELL_DATA           : 6,
104         TEXTURE_COORDINATES : 7,
105         SCALARS             : 8,
106         LOOKUP_TABLE        : 9,
107         COLOR_SCALARS       : 10
108     };
109     
110    this.json = {name:name};
111    
112    
113    function createLineIndices(line){
114        var count = line.length  -1;
115        if (count != line[0]){
116            throw 'Assertion Error. Inconsistent line';
117        }
118        var values = line.splice(1, count);
119        for (var i=0; i<count-1; i+=1){
120            indices.push(parseInt(values[i]));
121            indices.push(parseInt(values[i+1]));
122        }
123    } 
124     
125    for(var linenumber=0; linenumber<lines.length; linenumber++)
126    {
127     
128        try
129             {
130                 if (lines[linenumber].indexOf('POINTS') == 0)
131                 {
132                     console.log(lines[linenumber]);
133                     location = tags.POINTS;
134                     continue;
135                 }
136                 else if (lines[linenumber].indexOf('LINES') == 0)
137                 {
138                     console.log(lines[linenumber]);
139                     location = tags.LINES;
140                     mode = "LINES";
141                     continue;
142                 }
143                 else if (lines[linenumber].indexOf('TRIANGLE_STRIPS') == 0){
144                     console.log(lines[linenumber]);
145                     alert('vxlVTKParser ERROR: \n'+'Triangle Strips Not Supported. Please triangulate first');
146                     throw('Triangle Strips Not Supported. Please triangulate first');
147                 }
148                 else if (lines[linenumber].indexOf('POLYGONS')==0)
149                 {
150                     console.log(lines[linenumber]);
151                     location = tags.POLYGONS;
152                     continue;
153                 }
154                 else if (lines[linenumber].indexOf('POINT_DATA')==0)
155                 {                   
156                     location = tags.POINT_DATA;
157                     continue;
158                 }
159                 else if (lines[linenumber].indexOf('NORMALS')==0)
160                 {      
161                     console.log(lines[linenumber]);             
162                     location = tags.NORMALS;
163                     continue;
164                 }
165                 else if (lines[linenumber].indexOf('CELL_DATA')==0)
166                 {      
167                     console.log(lines[linenumber]);             
168                     location = tags.CELL_DATA;
169                     continue;
170                 }
171                 else if (lines[linenumber].indexOf('TEXTURE_COORDINATES')==0)
172                 {      
173                     console.log(lines[linenumber]);             
174                     location = tags.TEXTURE_COORDINATES;
175                     continue;
176                 }
177                 else if (lines[linenumber].indexOf('SCALARS')==0)
178                 {      
179                     console.log(lines[linenumber]);             
180                     location = tags.SCALARS;
181                     continue;
182                 }
183                 else if (lines[linenumber].indexOf('LOOKUP_TABLE')==0)
184                 {      
185                     console.log(lines[linenumber]);             
186                     location = tags.LOOKUP_TABLE;
187                     continue;
188                 }   
189                 else if (lines[linenumber].indexOf('COLOR_SCALARS')==0)
190                 {      
191                     console.log(lines[linenumber]);             
192                     location = tags.COLOR_SCALARS;
193                     continue;
194                 }           
195                 
196                 // -------------------
197                 else if(location == tags.POINTS)
198                 {
199                     var v = lines[linenumber].trim().split(' ');
200                     if (v == "") continue;
201                     for (var i=0; i<v.length; i++)
202                     {
203                          vertices.push(parseFloat(v[i]));
204                     }
205                 }
206                 else if(location == tags.LINES)
207                 {
208                     var elements = lines[linenumber].trim().split(' ');
209                     if (elements == "") continue;
210                     createLineIndices(elements);
211                 }
212                else if(location == tags.POLYGONS) //they have to be triangles
213                 {
214                     var tt = lines[linenumber].trim().split(' ');
215                     if (tt=="") continue; 
216                     if(tt.length>0 && tt[0] != '3')
217                     {
218                         throw "Error: please make sure your vtk file contains triangles instead of polygons (triangulate first)";
219                     }
220                     for(var i= 1; i<tt.length; i++)
221                     {
222                         indices.push(parseInt(tt[i]));
223                     }
224                 }
225                 else if(location == tags.LOOKUP_TABLE)
226                 {
227                     if(lines[linenumber].indexOf('LOOKUP_TABLE')==0)
228                         continue;
229                     else
230                     {
231                         var pd = lines[linenumber].trim().split(' ');
232                         if (pd=="") continue;
233                         for(var i=0; i<pd.length; i++)
234                         {
235                             scalars.push(parseFloat(pd[i]));
236                         }
237                    }
238                 }
239                 else if(location == tags.COLOR_SCALARS)
240                 {
241                     var n = lines[linenumber].trim().split(' ');
242                     if (n=="") continue;
243                     for(var i=0; i<n.length; i++)
244                     {
245                         colors.push(parseFloat(n[i]));
246                     }
247                 }
248                 else if(location == tags.NORMALS)
249                 {
250                     var n = lines[linenumber].trim().split(' ');
251                     if (n=="") continue;
252                     for(var i=0; i<n.length; i++)
253                     {
254                         normals.push(parseFloat(n[i]));
255                     }
256                 }
257            } 
258         catch(err)
259             {
260             console.log('Error while processing line '+ linenumber.toString());
261             throw err;
262             }
263     }
264     
265     
266     this.json.vertices = vertices;
267     this.json.mode     = mode;
268     if (indices.length >0)   this.json.indices = indices;
269     if (normals.length >0)   this.json.normals = normals;
270     if (colors.length  >0)   this.json.colors  = colors;
271     if (scalars.length >0)   this.json.scalars = scalars;
272     
273 };

276 vxlVTKReader.prototype.getModel = function(){
277     var model = new vxlModel(this.json.name, this.json);
278     this.json = null;
279     delete this.json;
280     return model;
281 };
