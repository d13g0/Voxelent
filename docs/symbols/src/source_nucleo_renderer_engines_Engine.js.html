 44 /*-------------------------------------------------------------------------*/
 45 // Engine Set Up
 46 /*-------------------------------------------------------------------------*/
 47 /**
 48  * Initializes the engine
 49  * @param {Object} p_renderer the renderer using this instance of the engine
 50  */
 51 vxlEngine.prototype.init = function(p_renderer){
 52     
 53     this._getGL(p_renderer);
 54     this._view       = p_renderer.view;
 55     this.pm          = new vxlProgramManager(this.gl);  
 56     this._transforms = new vxlTransforms(p_renderer.view);
 57      
 58     this.configure();
 59 };
 60 
 61 /**
 62  * Obtains the WebGL context
 63  */
 64 vxlEngine.prototype._getGL = function(p_renderer){
 65     var ctx        = undefined;
 66     var canvas     = p_renderer.view.canvas;
 67     var names      = ["webgl", "experimental-webgl", "webkit-3d", "moz-webgl"];
 68     
 69     this.gl        = undefined;
 70     
 71     for (var i = 0; i < names.length; ++i) {
 72         try {
 73             ctx = canvas.getContext(names[i]);
 74         } 
 75         catch(e) {}
 76         if (ctx) {
 77             break;
 78         }
 79     }
 80     if (ctx == null) {
 81         //@TODO: print a nicer jquery  alert
 82         alert("Sorry: WebGL is not available on this browser. Have you tried the newest version of Firefox, Chrome or Safari?"); 
 83         return undefined;
 84     }
 85     else {
 86         this.gl = ctx;
 87     }
 88 };
 89 
 90 /**
 91 * Configures some gl variables
 92 * 
 93 * 
 94 */
 95 vxlEngine.prototype.configure = function(){
 96     //@TODO: Review depth test and blending functions maybe these should be configurable
 97     var gl = this.gl;
 98     gl.enable(gl.DEPTH_TEST);
 99     gl.enable(gl.BLEND);
100     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
101     gl.depthFunc(gl.LESS);
102     gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
103  
104 };

106 /*-------------------------------------------------------------------------*/
107 // Clearing the canvas methods
108 /*-------------------------------------------------------------------------*/
109 /**
110  * Clears the canvas
111  */
112 vxlEngine.prototype.clear = function(){
113     var gl     = this.gl;
114     var canvas = this._view.canvas;
115     var width  = canvas.width;
116     var height = canvas.height;
117     var cc     = this._clear_color;
118     
119     gl.clearColor(cc[0], cc[1], cc[2], 1.0);
120     gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
121     gl.viewport(0, 0, width, height); //@TODO: Think about dividing view ports for multi-view apps - March 19/2012
122     
123     this._transforms.calculatePerspective();
124 };
125 
126 /**
127  * Sets the color used to clear the canvas
128  * @param {Number, Array, vec3} r it can be the red component, a 3-dimensional Array or a vec3 (glMatrix)
129  * @param {Number} g if r is a number, then this parameter corresponds to the green component
130  * @param {Number} b if r is a number, then this parameter corresponds to the blue component
131  * @see vxlView#setBackgroundColor
132  */
133 vxlEngine.prototype.clearColor = function(r,g,b){
134     var cc = vxl.util.createArr3(r,g,b);
135     this._clear_color = cc;
136     this.gl.clearColor(cc[0], cc[1], cc[2], 1.0);
137 };
138 
139 /**
140  * Sets the clear depth
141  * @param {Number} d the new clear depth
142  */
143 vxlEngine.prototype.clearDepth = function(d){
144     this.gl.clearDepth(d);
145 };
