 17 //Inheritance stuff
 18 vxlLambertStrategy.prototype = new vxlBasicStrategy(undefined)
 19 vxlLambertStrategy.prototype.constructor = vxlLambertStrategy;
 20 
 21 /**
 22  * 
 23  * Implements a rendering strategy that works with the vxl.def.glsl.lambert program. 
 24  * A rendering strategy allows decoupling rendering specific code (i.e. code that access and 
 25  * communicates with the program) from the actor.
 26  * @constructor
 27  * @class a vxlLambertStrategy object is selected by default as the strategy that an actor
 28  * will use for rendering.
 29  * 
 30  * @extends vxlBasicStrategy
 31  * @author Diego Cantor
 32  * 
 33  */
 34 function vxlLambertStrategy(renderer){
 35 	vxlBasicStrategy.call(this,renderer);
 36 }
 37 
 38 
 39 /**
 40  * Renders the actors one by one
 41  * @param {vxlScene} scene the scene to render
 42  */
 43 vxlLambertStrategy.prototype.render = function(scene){
 44 
 45     //Updates the perspective matrix and passes it to the program
 46     var r       = this.renderer;
 47     var trx     = r.transforms
 48     var prg     = r.prg;
 49     var glsl    = vxl.def.glsl;
 50     
 51     trx.calculatePerspective();
 52     trx.calculateModelView();
 53     prg.setUniform(glsl.PERSPECTIVE_MATRIX, trx.pMatrix);
 54     
 55     var elements = scene.actors.concat(scene.toys.list);
 56     var NUM = elements.length;
 57     
 58     if (scene.frameAnimation != undefined){
 59         scene.frameAnimation.update();
 60     }
 61 
 62     for(var i = 0; i < NUM; i+=1){
 63         this._renderActor(elements[i]);
 64     }
 65 };

 69 
 70 vxlLambertStrategy.prototype._renderActor = function(actor){
 71 
 72     if (!actor.visible) return; //Quick and simple
 73     
 74 	var model 	= actor.model;
 75     var buffers = this._gl_buffers[actor.UID]; 
 76     var r  		= this.renderer;
 77 	var gl 		= r.gl;
 78 	var prg 	= r.prg;
 79 	var trx 	= r.transforms;
 80 	var glsl    = vxl.def.glsl;
 81 	
 82 	this._applyActorTransform(actor);
 83 	
 84 	var diffuse = [actor.color[0], actor.color[1], actor.color[2],1.0];
 85 	
 86 	if (actor.opacity < 1.0){
 87 		diffuse[3] = actor.opacity;
 88 	}
 89 	else {
 90 		diffuse[3] = 1.0;
 91 	}
 92 	
 93 	prg.setUniform("uMaterialDiffuse",diffuse);
 94 	prg.setUniform("uUseVertexColors", false);
 95     
 96     prg.disableAttribute(glsl.COLOR_ATTRIBUTE);
 97 	prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
 98 	prg.enableAttribute(glsl.VERTEX_ATTRIBUTE);
 99 	
100 	try{
101 		
102         gl.bindBuffer(gl.ARRAY_BUFFER, buffers.vertex);
103         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices.slice(0)), gl.STATIC_DRAW);
104 		prg.setAttributePointer(glsl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
105 	}
106     catch(err){
107         alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
108 		throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
109     }
110     	
111 	if (actor.colors){	
112 		try{
113 			prg.setUniform("uUseVertexColors", true);
114 			gl.bindBuffer(gl.ARRAY_BUFFER, buffers.color);
115 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(actor.colors.slice(0)), gl.STATIC_DRAW);
116 			
117 			prg.enableAttribute(glsl.COLOR_ATTRIBUTE);
118 			prg.setAttributePointer(glsl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 0, 0);
119 		}
120 		catch(err){
121         	alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
122 			throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the color buffer. Error =' +err.description);
123    		}
124     }
125     
126     if(model.normals){
127 	    try{
128 			gl.bindBuffer(gl.ARRAY_BUFFER, buffers.normal);
129 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals.slice(0)), gl.STATIC_DRAW);
130 			
131 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
132 			prg.setAttributePointer(glsl.NORMAL_ATTRIBUTE,3,gl.FLOAT, false, 0,0);
133 		}
134 	    catch(err){
135 	        alert('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
136 			throw('There was a problem while rendering the actor ['+actor.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
137 	    }
138 	}
139     prg.setUniform("uUseShading",actor.shading);
140     
141     try{
142 		if (actor.mode == vxl.def.actor.mode.SOLID){
143 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
144 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
145 			gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
146 		}
147 		else if (actor.mode == vxl.def.actor.mode.WIREFRAME){
148 			if (actor.name == 'floor'){
149 			     prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
150 			}
151 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.wireframe);
152 			gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
153 			gl.drawElements(gl.LINES, model.wireframe.length, gl.UNSIGNED_SHORT,0);
154 		}
155 		else if (actor.mode == vxl.def.actor.mode.POINTS){
156 			prg.setUniform("uUseShading", true);
157 			prg.enableAttribute(glsl.NORMAL_ATTRIBUTE);
158 			gl.drawArrays(gl.POINTS,0, this.model.vertices.length/3);
159 		}
160 		else if (actor.mode == vxl.def.actor.mode.LINES){
161 			prg.setUniform("uUseShading", false);
162 			prg.disableAttribute(glsl.NORMAL_ATTRIBUTE);
163 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffers.index);
164 			gl.drawElements(gl.LINES, actor.model.indices.length, gl.UNSIGNED_SHORT,0);
165 		}
166 		else{
167             alert('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+this.mode+' is not valid.');
168 			throw('There was a problem while rendering the actor ['+actor.name+']. The visualization mode: '+this.mode+' is not valid.');
169 			
170 		}
171 		//Cleaning up
172 		 gl.bindBuffer(gl.ARRAY_BUFFER, null);
173 	     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
174     }
175 	catch(err){
176 		alert('Error rendering actor ['+actor.name+']. Error =' +err.description);
177 		throw('Error rendering actor ['+actor.name+']. Error =' +err.description);
178 	}
179 };
