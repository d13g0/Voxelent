 22 /**
 23  * @class
 24  * @constructor
 25  */
 26 function vxlActor(model){
 27   
 28   
 29   
 30   this.bb = []
 31   this.allocated = false;
 32   this.visible   = true;
 33   this.mode = vxl.def.actor.mode.SOLID;
 34   this.opacity = 1.0;
 35   this.colors = model?(model.colors!=null?model.colors:null):null;	//it will create colors for this actor once a lookup table had been set up
 36   
 37   this.position 	= vec3.create([0, 0, 0]);
 38   this.scale 		= vec3.create([1, 1, 1]);
 39   this.rotation 	= vec3.create([0, 0, 0]);
 40   
 41   this.program       = undefined;
 42   this.picking_color = undefined;
 43   
 44   this.renderers = [];
 45   this.buffers = [];
 46   this.clones  = 0;
 47   
 48   if (model){
 49   	this.model 	 = model;
 50   	this.name 	 = model.name;
 51   	this.diffuse = model.diffuse;
 52   	this.bb 	 = model.outline;
 53   	this.mode    = model.mode==undefined?vxl.def.actor.mode.SOLID:model.mode;
 54   }
 55   
 56 };

 99 vxlActor.prototype.getPosition = function(){
100 	cc = this.centre;
101 	bb = this.bb;
102 	
103 	cc[0] = (bb[3] + bb[0]) /2;
104 	cc[1] = (bb[4] + bb[1]) /2;
105 	cc[2] = (bb[5] + bb[2]) /2;
106 		
107 	cc[0] = Math.round(cc[0]*1000)/1000;
108 	cc[1] = Math.round(cc[1]*1000)/1000;
109 	cc[2] = Math.round(cc[2]*1000)/1000;
110 	
111 	return cc;
112 };
113 /**
114 * Creates the internal WebGL buffers that will store geometry, normals, colors, etc for this Actor.
115 * It uses the renderer passed as a parameter to retrieve the gl context to use.
116 */
117 vxlActor.prototype.allocate = function(renderer){
118 	
119 	//if (this.allocated) return; // if we need realocation create method reallocate to force it.
120 	
121 	//as we don't expect changes we set the buffers' data here.
122    //OTHERWISE it should be done in the draw method as it is done with the axis and the bounding box
123    
124    if (this.renderers.indexOf(renderer)!=-1){ //if this renderer has been allocated then ignore
125    		return;
126    }
127    
128    vxl.go.console('Actor: Allocating actor '+this.name+' for view '+ renderer.view.name);
129    	
130 	var gl = renderer.gl;
131 	var model = this.model;
132     var buffer = {};
133 	
134 	buffer.vertex = gl.createBuffer();
135 	gl.bindBuffer(gl.ARRAY_BUFFER, buffer.vertex);
136 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices), gl.STATIC_DRAW);
137 	
138 	if (model.normals){
139 		buffer.normal = gl.createBuffer();
140 		gl.bindBuffer(gl.ARRAY_BUFFER, buffer.normal);
141 		gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals), gl.STATIC_DRAW);
142 	}
143 	
144 	if (model.scalars != undefined || model.colors != undefined){
145 		buffer.color = gl.createBuffer(); //we don't BIND values or use the buffer until the lut is loaded and available
146 	}
147 	
148 	
149 	
150 	if (model.wireframe != undefined){
151 		buffer.wireframe = gl.createBuffer();
152 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer.wireframe);
153 		gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
154 	}
155 	
156 	if (model.indices != undefined){
157 		buffer.index = gl.createBuffer();
158 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer.index);
159 		gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.indices), gl.STATIC_DRAW);
160 	}
161 	
162 	gl.bindBuffer(gl.ARRAY_BUFFER, null);
163 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
164 	
165 	this.renderers.push(renderer);
166 	this.buffers.push(buffer);
167 };

179 /**
180  * Passes the matrices to the shading program
181  * @param {vxlRenderer} renderer determines the context for the transformations, 
182  * different renderers can have different matrices transformations 
183  * we will update each Model-View matrix of each renderer according to
184  * the actor position,scale and rotation.
185  */
186 vxlActor.prototype.updateMatrixStack = function(renderer){
187     
188     var r	= renderer,	trx = r.transforms,	prg = r.prg;
189     trx.calculateModelView();
190     trx.push();
191         mat4.scale      (trx.mvMatrix, this.scale);
192 		mat4.translate	(trx.mvMatrix, this.position);
193 		//TODO: Implement actor rotations
194 	    prg.setUniform("uMVMatrix",r.transforms.mvMatrix);
195     trx.pop();
196     trx.calculateNormal(); 
197     prg.setUniform("uPMatrix", r.transforms.pMatrix);
198     prg.setUniform("uNMatrix", r.transforms.nMatrix);
199     
200  };

202 /**
203 * Performs the rendering of this actor using the WebGL context provided by the renderer
204 * This method has the most WebGL code in all Nucleo.
205 * 
206 */
207 vxlActor.prototype.render = function(renderer){
208 	
209 	
210 	if (!this.visible){ 
211 		return;
212 	}
213 
214 	//if (this.program){
215 	  //renderer.setProgram(this.program);
216 	//} 
217 	//else {
218 	  //  renderer.setProgram(renderer.defaultProgram);
219 //	}
220 	
221 	
222 	var idx = this.renderers.indexOf(renderer);
223 
224 	var model = this.model;
225     var buffer = this.buffers[idx]; 
226 	
227     //The actor is a good renderer friend.
228 	var gl = renderer.gl;
229 	var prg = renderer.prg;
230 	var trx = renderer.transforms;
231 	
232 	//First thing first. Handle actor transformations here
233 	this.updateMatrixStack(renderer);
234 	
235 	if (this.opacity < 1.0){
236 		gl.disable(gl.DEPTH_TEST);
237 		gl.enable(gl.BLEND);
238 		this.diffuse[3] = this.opacity;
239 	}
240 	else {
241 		gl.enable(gl.DEPTH_TEST);
242 		gl.disable(gl.BLEND);
243 		this.diffuse[3] = 1.0;
244 	}
245 	
246 	prg.setUniform("uMaterialDiffuse",this.diffuse);
247 	prg.setUniform("uUseVertexColors", false);
248     
249     prg.disableAttribute("aVertexColor");
250 	prg.disableAttribute("aVertexNormal");
251 	prg.enableAttribute("aVertexPosition");
252 	try{
253 		
254         gl.bindBuffer(gl.ARRAY_BUFFER, buffer.vertex);
255         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.vertices.slice(0)), gl.STATIC_DRAW);
256 		prg.setAttributePointer("aVertexPosition", 3, gl.FLOAT, false, 0, 0);
257 	}
258     catch(err){
259         alert('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
260 		throw('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the vertex buffer. Error =' +err.description);
261     }
262     	
263 	if (this.colors){	
264 		try{
265 			prg.setUniform("uUseVertexColors", true);
266 			gl.bindBuffer(gl.ARRAY_BUFFER, buffer.color);
267 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(this.colors.slice(0)), gl.STATIC_DRAW);
268 			
269 			prg.enableAttribute("aVertexColor");
270 			prg.setAttributePointer("aVertexColor", 3, gl.FLOAT, false, 0, 0);
271 		}
272 		catch(err){
273         	alert('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the color buffer. Error =' +err.description);
274 			throw('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the color buffer. Error =' +err.description);
275    		}
276     }
277     
278     if(model.normals){
279 	    try{
280 			gl.bindBuffer(gl.ARRAY_BUFFER, buffer.normal);
281 			gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(model.normals.slice(0)), gl.STATIC_DRAW);
282 			
283 			prg.enableAttribute("aVertexNormal");
284 			prg.setAttributePointer("aVertexNormal",3,gl.FLOAT, false, 0,0);
285 		}
286 	    catch(err){
287 	        alert('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
288 			throw('There was a problem while rendering the actor ['+this.name+']. The problem happened while handling the normal buffer. Error =' +err.description);
289 	    }
290 	}
291     
292     try{
293 		if (this.mode == vxl.def.actor.mode.SOLID){
294 			prg.setUniform("uUseShading",true);
295 			prg.enableAttribute("aVertexNormal");
296 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer.index);
297 			gl.drawElements(gl.TRIANGLES, model.indices.length, gl.UNSIGNED_SHORT,0);
298 		}
299 		else if (this.mode == vxl.def.actor.mode.WIREFRAME){
300 			prg.setUniform("uUseShading", false);
301 			if (this.name == 'floor'){
302 			     prg.disableAttribute("aVertexNormal");
303 			}
304 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer.wireframe);
305 			gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(model.wireframe), gl.STATIC_DRAW);
306 			gl.drawElements(gl.LINES, model.wireframe.length, gl.UNSIGNED_SHORT,0);
307 		}
308 		else if (this.mode == vxl.def.actor.mode.POINTS){
309 			prg.setUniform("uUseShading", true);
310 			prg.enableAttribute("aVertexNormal");
311 			gl.drawArrays(gl.POINTS,0, this.model.vertices.length/3);
312 		}
313 		else if (this.mode == vxl.def.actor.mode.LINES){
314 			prg.setUniform("uUseShading", false);
315 			prg.disableAttribute("aVertexNormal");
316 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, buffer.index);
317 			gl.drawElements(gl.LINES, this.model.indices.length, gl.UNSIGNED_SHORT,0);
318 		
319 		}
320 		else{
321             alert('There was a problem while rendering the actor ['+this.name+']. The visualization mode: '+this.mode+' is not valid.');
322 			throw('There was a problem while rendering the actor ['+this.name+']. The visualization mode: '+this.mode+' is not valid.');
323 			
324 		}
325 		 gl.bindBuffer(gl.ARRAY_BUFFER, null);
326 	     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
327 	     
328 
329     }
330 	catch(err){
331 		alert('Error rendering actor ['+this.name+']. Error =' +err.description);
332 		throw('Error rendering actor ['+this.name+']. Error =' +err.description);
333 	}
334 	
335 };

399 /**
400  * Duplicates this actor.
401  * 
402  * Properties copied by REFERENCE:
403  * model,
404  * buffers,
405  * renderers
406  * 
407  * Everything else is copied by VALUE.
408  * 
409  * This method is fundamental to replicate objects in the scene, without having to duplicate
410  * the shared model. A cloned actor however can have different position, colors, properties, etc.
411  * 
412  * If a cloned actor modifies his internal model, any other actor that shares the model will be
413  * affected. 
414  * 
415  * The returned actor is not added to the scene automatically. It is up the
416  * programmer to determine the scene the cloned actor needs to be added to if any.
417  * 
418  * @see vxlModel
419  * @returns {vxlActor} an actor 
420  */
421 vxlActor.prototype.clone = function(){
422     this.clones++;
423 	
424 	var duplicate = new vxlActor(this.model);
425 	duplicate['program']   = this['program'];
426 	vec3.set(this.scale,    duplicate.scale);
427 	vec3.set(this.position, duplicate.position);
428 	
429 	
430 	//Now to save us some memory, let's SHARE the WebGL buffers that the current actor has already allocated'
431 	//duplicate.renderers = this.renderers;
432 	//duplicate.buffers   = this.buffers;
433 	//duplicate.model 	= this.model;
434 	duplicate.name     += '-'+this.clones; 
435 	return duplicate;
436 };
