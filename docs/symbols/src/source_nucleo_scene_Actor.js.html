 23 /**
 24  * <p>
 25  * An actor is a representation of a model in Voxelent. Actors can cache model properties 
 26  * and modified them. This is useful when there are several actors based in the same model
 27  * but each one of them needs to have a different version of any given model property (i.e. color)
 28  * </p>
 29  * <p> 
 30  * To propagate one change for all the actors based in the same model, the setProperty method
 31  * should be invoked by setting the third parameter (scope) like this:
 32  * </p>
 33  * 
 34  * <pre class="prettyprint">
 35  * var actor = vxl.c.scene.getActorByName('example.json');
 36  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.model)
 37  * </pre>
 38  * 
 39  * <p>If the change should be local (for just that actor)  then you should write:</p>
 40  * 
 41  * <pre class="prettyprint">
 42  * var actor = vxl.c.scene.getActorByName('example.json');
 43  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.actor)
 44  * </pre>
 45  * 
 46  * <p> Or simply </p>
 47  *  
 48  * <pre class="prettyprint">
 49  * var actor = vxl.c.scene.getActorByName('example.json');
 50  * actor.setProperty('color',[1.0,0.0,0.0])
 51  * </pre>
 52  * @class Actors represent models (assets) in a Scene
 53  * @constructor
 54  */
 55 function vxlActor(model){
 56   
 57   this.bb = [0, 0, 0, 0, 0, 0];
 58   this.position 	= vec3.create([0, 0, 0]);
 59   this.centre       = vec3.create([0, 0, 0]);
 60   this.scale 		= vec3.create([1, 1, 1]);
 61   this.rotation 	= vec3.create([0, 0, 0]);
 62 
 63   this.visible      = true;
 64   this.mode         = vxl.def.actor.mode.SOLID;
 65   this.opacity      = 1.0;
 66   this.colors       = model?(model.colors!=null?model.colors:null):null;	//it will create colors for this actor once a lookup table had been set up
 67   this.clones       = 0;
 68   
 69   this._renderers   = [];
 70   this._gl_buffers  = [];
 71   this.shading      = true;
 72   
 73   this.scene        = undefined;
 74 
 75   
 76   if (model){
 77   	this.model 	    = model;
 78   	this.name 	    = model.name;
 79   	//In the newest versions of Voxelent JSON format, the diffuse property has been replaced with color property.
 80   	this.color      = model.color!=undefined?model.color:(model.diffuse!=undefined?model.diffuse:undefined); 
 81   	this.bb 	    = model.bb.slice(0);
 82   	this.mode       = model.mode==undefined?vxl.def.actor.mode.SOLID:model.mode;
 83   	this.centre     = vec3.set(model.centre, vec3.create());
 84   }
 85   
 86   var e = vxl.events;
 87   vxl.go.notifier.publish(
 88       [
 89         e.ACTOR_MOVED,
 90         e.ACTOR_SCALED,
 91         e.ACTOR_CHANGED_COLOR,
 92         e.ACTOR_CHANGED_SHADING
 93       ], this);
 94   
 95   
 96   this.UID = vxl.util.generateUID();
 97 };

109 /**
110  * Sets the position of this actor. 
111  * 
112  * It updates the bounding box according to the position of the actor. The position of the actor
113  * is initially [0,0,0] and it is relative to the model centre. If the model is not centered in 
114  * the origin, then the actor's position will be relative to the model centre.
115  * 
116  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
117  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
118  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
119  */
120 vxlActor.prototype.setPosition = function (x,y,z){
121     this.position = vxl.util.createVec3(x,y,z); 
122     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
123 };

126 /**
127  * Scales this actor. 
128  * @param {Number, Array, vec3} s the scaling factor. The scaling factor is applied in all axes.
129  *
130  */
131 vxlActor.prototype.setScale = function(s,a,b){
132     if (s == 0 && a == undefined && b==undefined) return;
133     
134     
135     if (typeof(s)=="number" && a == undefined && b == undefined){
136         this.scale = vxl.util.createVec3(s,s,s);
137     }
138     else{
139         this.scale = vxl.util.createVec3(s,a,b);
140     }
141     vxl.go.notifier.fire(vxl.events.ACTOR_SCALED, this);
142 };

144 /**
145  * Computes the current bounding box for this actor.
146  * This method is called on demand by the scene or any other object. 
147  * The bounding box is NOT automatically recalculated when 
148  * moving or scaling an actor for performance reasons.
149  *  
150  */
151 vxlActor.prototype.computeBoundingBox = function(){
152 
153     this.bb  = this.model.bb.slice(0);
154 
155     var bb = this.bb;
156     var cc = this.model.centre;
157 
158     var pos =this.position;
159     
160     //Calculate position
161     bb[0] += pos[0];
162     bb[1] += pos[1];
163     bb[2] += pos[2];
164     bb[3] += pos[0];
165     bb[4] += pos[1];
166     bb[5] += pos[2];
167     
168     var minY = bb[1];
169 
170     //Recalculate centre 
171     cc[0] = (bb[3] + bb[0]) /2;
172     cc[1] = (bb[4] + bb[1]) /2;
173     cc[2] = (bb[5] + bb[2]) /2;
174     
175     //Calculate scale
176     bb[0] *= this.scale[0];
177     bb[1] *= this.scale[1];
178     bb[2] *= this.scale[2];
179     bb[3] *= this.scale[0];
180     bb[4] *= this.scale[1];
181     bb[5] *= this.scale[2];
182     
183     //Now recenter
184     var c = [0,0,0];
185     c[0] = (bb[3] + bb[0]) /2;
186     c[1] = (bb[4] + bb[1]) /2;
187     c[2] = (bb[5] + bb[2]) /2;
188     
189     //recalculate boundingbox
190     var shift = [c[0]-cc[0], bb[1]-minY,c[2]-cc[2]];
191     bb[0] -= shift[0];
192     bb[1] -= shift[1];
193     bb[2] -= shift[2];
194     bb[3] -= shift[0];
195     bb[4] -= shift[1];
196     bb[5] -= shift[2];
197 };

201 * @param {Number, Array, vec3} r it can be the red component, a 3-dimensional Array or a vec3 (glMatrix)

224 /**
225  * If the property exists, then it updates it
226  * @param {String} property 
227  * @param {Object} value 
228  * @param {String} scope indicates if the change is made at the actor level or at the model level
229  * valid values for scope are vxl.def.model and vxl.def.actor
230  * @TODO: if the property is position or scale then call the respective methods from here
231  */
232 vxlActor.prototype.setProperty = function(property, value, scope){
233     
234     if (property == 'position') {
235     	this.setPosition(value);
236     	return;
237     }
238     
239     if (property == 'scale')    {
240     	this.setScale(value);
241     	return;
242     }
243     
244     if (property == 'color') {
245         this.setColor(value);
246         return;
247     }
248     
249     if (property == 'shading'){
250         this.setShading(value);
251         return;
252     }
253     
254     if (scope == vxl.def.actor || scope == undefined || scope == null){
255 		this[property] = value;
256 		vxl.go.console('Actor: The actor '+this.name+' has been updated. ['+property+' = '+value+']');
257 	}
258 	else if(scope == vxl.def.model){
259 		this.model[property] = value;
260 	}
261 	else{
262 		throw('vxlActor.setProperty. Scope:' + scope +' is not valid');
263 	}
264 	
265 };

278 /**
279  * Returns an actor property if that property exists in the actor. Otherwise it will search 
280  * in the model. This method is used by the renderer. There are some cases where actors have local changes
281  * that are not reflected in the model. In these cases the renderer should pick the actor property
282  * over the model property
283  * @param {String} property the property name
284  * @returns {Object} the property or undefined if the property is not found 
285  */
286 vxlActor.prototype.getProperty = function(property){
287 	if (this.hasOwnProperty(property)) {
288 		return this[property];
289 	}
290 	else if (this.model.hasOwnProperty(property)){
291 		return this.model[property];
292 	}
293 	else {
294 		return undefined;
295 	}
296 };

298 /**
299  * Estimates the current position as
300  * the center of the current bounding box. 
301  * This method does not update the internal position of the actor
302  * it only returns an estimate based on the location of its bounding box.
303  */
304 vxlActor.prototype.computePosition = function(){
305 	bb = this.bb;
306 	var cc = this.position;
307 	
308 	cc[0] = (bb[3] + bb[0]) /2;
309 	cc[1] = (bb[4] + bb[1]) /2;
310 	cc[2] = (bb[5] + bb[2]) /2;
311 		
312 	cc[0] = Math.round(cc[0]*1000)/1000;
313 	cc[1] = Math.round(cc[1]*1000)/1000;
314 	cc[2] = Math.round(cc[2]*1000)/1000;
315 	
316 	return vec3.create(cc); 
317 };

329 /**
330 * Sets the lookup table for this actor.
331 * This method will only succeed if the model that this actor represents has scalars 
332 * @param {String} lutID the lookup table id. See {@link vxl.def.lut.list} for currently supported ids.
333 * @param {Number} min lowest value for interpolation
334 * @param {Number} max highest value for interpolation
335 */
336 vxlActor.prototype.setLookupTable = function(lutID,min,max){
337 	if (this.model.scalars){
338 		var lut = vxl.go.lookupTableManager.get(lutID);
339 		this.colors  = lut.getColors(this.model.scalars,min,max);
340 	}
341 };

367 /**
368  * Duplicates this actor.
369  * 
370  * Properties copied by REFERENCE:
371  * model,
372  * buffers,
373  * renderers
374  * 
375  * Everything else is copied by VALUE.
376  * 
377  * This method is fundamental to replicate objects in the scene, without having to duplicate
378  * the shared model. A cloned actor however can have different position, colors, properties, etc.
379  * 
380  * If a cloned actor modifies his internal model, any other actor that shares the model will be
381  * affected. 
382  * 
383  * The returned actor is not added to the scene automatically. It is up the
384  * programmer to determine the scene the cloned actor needs to be added to if any.
385  * 
386  * @see vxlModel
387  * @returns {vxlActor} an actor 
388  */
389 vxlActor.prototype.clone = function(){
390     this.clones++;
391 	
392 	var duplicate = new vxlActor(this.model);
393 	duplicate.setScene(this.scene);
394 	
395 	vec3.set(this.scale,    duplicate.scale);
396 	vec3.set(this.position, duplicate.position);
397 	
398 	
399 	//Now to save us some memory, let's SHARE the WebGL buffers that the current actor has already allocated'
400 	//duplicate.renderers = this._renderers;
401 	//duplicate.buffers   = this._gl_buffers;
402 
403 	duplicate.name     += '-'+this.clones; 
404 	return duplicate;
405 };
