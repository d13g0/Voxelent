 23 /**
 24  * <p>
 25  * An actor is a representation of a model in Voxelent. Actors can cache model properties 
 26  * and modified them. This is useful when there are several actors based in the same model
 27  * but each one of them needs to have a different version of any given model property (i.e. color)
 28  * </p>
 29  * <p> 
 30  * To propagate one change for all the actors based in the same model, the setProperty method
 31  * should be invoked by setting the third parameter (scope) like this:
 32  * </p>
 33  * 
 34  * <pre class="prettyprint">
 35  * var actor = vxl.c.scene.getActorByName('example');
 36  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.model)
 37  * </pre>
 38  * 
 39  * <p>If the change should be local (for just that actor)  then you should write:</p>
 40  * 
 41  * <pre class="prettyprint">
 42  * var actor = vxl.c.scene.getActorByName('example');
 43  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.actor)
 44  * </pre>
 45  * 
 46  * <p> Or simply </p>
 47  *  
 48  * <pre class="prettyprint">
 49  * var actor = vxl.c.scene.getActorByName('example');
 50  * actor.setProperty('color',[1.0,0.0,0.0])
 51  * </pre>
 52  * @class Actors represent models (assets) in a Scene
 53  * @constructor
 54  */
 55 function vxlActor(model){
 56   
 57   this._bb = [0, 0, 0, 0, 0, 0];
 58   this._position 	= vec3.create([0, 0, 0]);
 59   this._translation = vec3.create([0, 0, 0]);
 60   this._centre      = vec3.create([0, 0, 0]);
 61   this._scale 		= vec3.create([1, 1, 1]);
 62   this._rotation 	= vec3.create([0, 0, 0]);
 63   this._matrix      = mat4.identity();
 64 
 65   this.visible      = true;
 66   this.mode         = vxl.def.actor.mode.SOLID;
 67   this.opacity      = 1.0;
 68   this.colors       = model?(model.colors!=null?model.colors:null):null;	//it will create colors for this actor once a lookup table had been set up
 69   this.clones       = 0;
 70   
 71   this._renderers   = [];
 72   this._gl_buffers  = [];
 73   this.shading      = true;
 74   this.texture      = undefined;
 75   
 76   this.scene        = undefined;
 77   
 78   this._trackingCameras  = [];
 79 
 80   
 81   if (model){
 82   	this.model 	    = model;
 83   	this.name 	    = model.name;
 84   	//In Voxelent JSON format (0.87), the diffuse property has been replaced with color property.
 85   	this.color      = model.color!=undefined?model.color:(model.diffuse!=undefined?model.diffuse:undefined); 
 86   	this.mode       = model.mode;
 87   	this._bb         = model.bb.slice(0);
 88   	this._centre     = vec3.set(model.centre, vec3.create());
 89   }
 90   
 91   if (this.mode == vxl.def.actor.mode.TEXTURED){
 92       this.setTexture(this.model.path + this.model.texture);
 93   }
 94   
 95   
 96   
 97   var e = vxl.events;
 98   vxl.go.notifier.publish(
 99       [
100         e.ACTOR_MOVED,
101         e.ACTOR_SCALED,
102         e.ACTOR_ROTATED,
103         e.ACTOR_CHANGED_COLOR,
104         e.ACTOR_CHANGED_SHADING
105       ], this);
106   
107   
108   this.UID = vxl.util.generateUID();
109 };

121 /**
122  * Sets the position of this actor. 
123  * 
124  * 
125  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
126  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
127  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
128  */
129 vxlActor.prototype.setPosition = function (x,y,z){
130     var np = vxl.util.createVec3(x,y,z);
131     vec3.subtract(np, this._position, this._translation);
132     this._position = np;  
133     
134     var m = this._matrix;
135     m[12] = this._position[0];
136     m[13] = this._position[1];
137     m[14] = this._position[2];
138     m[15] = 1;
139     this._computeBoundingBox();
140     this._notifyTrackingCameras();
141     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
142 };

145 /**
146  * Translates the actor by a given vector 
147  * 
148  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
149  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
150  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
151  */
152 vxlActor.prototype.translate = function (x,y,z){
153     this._translation = vxl.util.createVec3(x,y,z); 
154     
155     var m = this._matrix;
156     mat4.translate(m,this._translation);
157     this._position = vec3.createFrom(m[12], m[13], m[14]);
158     this._computeBoundingBox();
159     this._notifyTrackingCameras();
160     vxl.go.notifier.fire(vxl.events.ACTOR_MOVED, this);
161 };

204 /**
205  * Scales this actor. 
206  * @param {Number, Array, vec3} s the scaling factor. The scaling factor is applied in all axes.
207  *
208  */
209 vxlActor.prototype.setScale = function(s,a,b){
210     if (s == 0 && a == undefined && b==undefined) return;
211     
212     
213     if (typeof(s)=="number" && a == undefined && b == undefined){
214         this._scale = vxl.util.createVec3(s,s,s);
215     }
216     else{
217         this._scale = vxl.util.createVec3(s,a,b);
218     }
219     
220     var m = this._matrix;
221     mat4.scale(m,this._scale);
222     this._computeBoundingBox();
223     vxl.go.notifier.fire(vxl.events.ACTOR_SCALED, this);
224 };

226 /**
227  * Adds a tracking camera
228  * @param{vxlCamera} camera the tracking camera
229  */
230 vxlActor.prototype.addTrackingCamera = function(camera){
231     if (camera instanceof vxlCamera && camera.type !=vxl.def.camera.type.TRACKING){
232         throw "vxlActor._addTrackingCamera ERROR: the selected camera is not set to tracking"
233     }
234     else if (camera instanceof vxlCamera){
235         this._trackingCameras.push(camera);
236     }
237     else{
238         throw "vxlActor._addTrackingCamera ERROR: the object passed as a parameter is not a vxlCamera"
239     }
240 };

242 /**
243  * Removes a tracking camera
244  * @param{vxlCamera} camera the tracknig camera
245  */
246 vxlActor.prototype.removeTrackingCamera = function(camera){
247     var idx = this._trackingCameras.indexOf(camera);
248     this._trackingCameras.splice(idx,1);
249 };

251 /**
252  * Notifies all the tracking cameras of the actor translation 
253  * @param {Object} camera
254  */
255 vxlActor.prototype._notifyTrackingCameras = function(){
256     for (var i=0, N = this._trackingCameras.length; i<N;i+=1){
257         this._trackingCameras[i].updateWithActor(this);
258     }
259 };

261 /**
262  * Computes the current bounding box for this actor.
263  * This method is called on demand by the scene or any other object. 
264  * The bounding box is NOT automatically recalculated when 
265  * moving or scaling an actor for performance reasons.
266  *  
267  */
268 vxlActor.prototype._computeBoundingBox = function(){
269 
270     var vs  = this.model.vertices;
271     var vsT = [];
272     var T = this._matrix;
273     
274     for(var i=0;i<vs.length;i=i+3){
275         var x = vxl.util.createVec3(vs[i],vs[i+1],vs[i+2]);
276         mat4.multiplyVec3(T, x);
277         vsT.push(x[0], x[1], x[2]);    
278     }
279     
280     var bbA = [vsT[0],vsT[1],vsT[2],vsT[0],vsT[1],vsT[2]];
281     
282     for(var i=0;i<vsT.length;i=i+3){
283         bbA[0] = Math.min(bbA[0],vsT[i]);
284         bbA[1] = Math.min(bbA[1],vsT[i+1]);
285         bbA[2] = Math.min(bbA[2],vsT[i+2]);
286         bbA[3] = Math.max(bbA[3],vsT[i]);
287         bbA[4] = Math.max(bbA[4],vsT[i+1]);
288         bbA[5] = Math.max(bbA[5],vsT[i+2]);
289     }
290     
291      
292     this._bb = bbA;
293 };

295 /**
296  * Returns an array with the bounding box vertices. Good for rendering the transformed
297  * bounding box (after geometric transformations) 
298  * @returns {Array} a 8-element array with the vertices that constitute the actor bounding box
299  */
300 vxlActor.prototype.getBoundingBoxVertices = function(){
301     var b = this._bb;
302     return [
303         b[0], b[1], b[2],
304         b[0], b[4], b[2],
305         b[3], b[4], b[2],
306         b[3], b[1], b[2],
307         b[0], b[1], b[5],
308         b[0], b[4], b[5],
309         b[3], b[4], b[5],
310         b[3], b[1], b[5] 
311         ];
312 };

369 /**
370  * If the property exists, then it updates it
371  * @param {String} property 
372  * @param {Object} value 
373  * @param {String} scope indicates if the change is made at the actor level or at the model level
374  * valid values for scope are vxl.def.model and vxl.def.actor
375  * @TODO: if the property is position or scale then call the respective methods from here
376  */
377 vxlActor.prototype.setProperty = function(property, value, scope){
378     
379     if (property == 'position') {
380     	this.setPosition(value);
381     	return;
382     }
383     
384     if (property == 'scale')    {
385     	this.setScale(value);
386     	return;
387     }
388     
389     if (property == 'color') {
390         this.setColor(value);
391         return;
392     }
393     
394     if (property == 'shading'){
395         this.setShading(value);
396         return;
397     }
398     
399     if (property == 'texture'){
400         this.setTexture(value);
401         return
402     }
403     
404     if (scope == vxl.def.actor || scope == undefined || scope == null){
405 		this[property] = value;
406 		vxl.go.console('Actor: The actor '+this.name+' has been updated. ['+property+' = '+value+']');
407 	}
408 	else if(scope == vxl.def.model){
409 		this.model[property] = value;
410 	}
411 	else{
412 		throw('vxlActor.setProperty. Scope:' + scope +' is not valid');
413 	}
414 	
415 };

428 /**
429  * Returns an actor property if that property exists in the actor. Otherwise it will search 
430  * in the model. This method is used by the renderer. There are some cases where actors have local changes
431  * that are not reflected in the model. In these cases the renderer should pick the actor property
432  * over the model property
433  * @param {String} property the property name
434  * @returns {Object} the property or undefined if the property is not found 
435  */
436 vxlActor.prototype.getProperty = function(property){
437 	if (this.hasOwnProperty(property)) {
438 		return this[property];
439 	}
440 	else if (this.model.hasOwnProperty(property)){
441 		return this.model[property];
442 	}
443 	else {
444 		return undefined;
445 	}
446 };

448 /**
449  * Estimates the current position as
450  * the center of the current bounding box. 
451  * This method does not update the internal position of the actor
452  * it only returns an estimate based on the location of its bounding box.
453  */
454 vxlActor.prototype.computePosition = function(){
455 	bb = this._bb;
456 	var cc = this._position;
457 	
458 	cc[0] = (bb[3] + bb[0]) /2;
459 	cc[1] = (bb[4] + bb[1]) /2;
460 	cc[2] = (bb[5] + bb[2]) /2;
461 		
462 	cc[0] = Math.round(cc[0]*1000)/1000;
463 	cc[1] = Math.round(cc[1]*1000)/1000;
464 	cc[2] = Math.round(cc[2]*1000)/1000;
465 	
466 	return vec3.create(cc); 
467 };

517 /**
518  * Duplicates this actor.
519  * 
520  * Properties copied by REFERENCE:
521  * model,
522  * buffers,
523  * renderers
524  * 
525  * Everything else is copied by VALUE.
526  * 
527  * This method is fundamental to replicate objects in the scene, without having to duplicate
528  * the shared model. A cloned actor however can have different position, colors, properties, etc.
529  * 
530  * If a cloned actor modifies his internal model, any other actor that shares the model will be
531  * affected. 
532  * 
533  * The returned actor is not added to the scene automatically. It is up the
534  * programmer to determine the scene the cloned actor needs to be added to if any.
535  * 
536  * @see vxlModel
537  * @returns {vxlActor} an actor 
538  */
539 vxlActor.prototype.clone = function(){
540     this.clones++;
541 	
542 	var duplicate = new vxlActor(this.model);
543 	duplicate.setScene(this.scene);
544 	
545 	//TODO: duplicate.setMatrix(this._matrix); this method should update all the other variables (position, scale, rot, etc).
546      	
547 	//Now to save us some memory, let's SHARE the WebGL buffers that the current actor has already allocated'
548 	//duplicate.renderers = this._renderers;
549 	//duplicate.buffers   = this._gl_buffers;
550 
551 	duplicate.name     += '-'+this.clones; 
552 	return duplicate;
553 };
