 22 /**
 23  * <p>
 24  * An actor is a representation of a model in voxelent. Actors can cache model properties 
 25  * and modified them. This is useful when there are several actors based in the same model
 26  * but each one of them needs to have a different version of any given model property (i.e. color)
 27  * </p>
 28  * <p> 
 29  * To propagate one change for all the actors based in the same model, the setProperty method
 30  * should be invoked by setting the third parameter (scope) like this:
 31  * </p>
 32  * 
 33  * <pre>
 34  * var actor = vxl.c.scene.getActorByName('example.json');
 35  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.model)
 36  * </pre>
 37  * 
 38  * <p>If the change should be local (for just that actor)  then you should write:</p>
 39  * 
 40  * <pre>
 41  * var actor = vxl.c.scene.getActorByName('example.json');
 42  * actor.setProperty('color',[1.0,0.0,0.0], vxl.def.actor)
 43  * </pre>
 44  * 
 45  * <p> Or simply </p>
 46  *  
 47  * <pre>
 48  * var actor = vxl.c.scene.getActorByName('example.json');
 49  * actor.setProperty('color',[1.0,0.0,0.0])
 50  * </pre>
 51  * @class
 52  * @constructor
 53  */
 54 function vxlActor(model){
 55   
 56   this.bb = [0, 0, 0, 0, 0, 0];
 57   this.position 	= vec3.create([0, 0, 0]);
 58   this.scale 		= vec3.create([1, 1, 1]);
 59   this.rotation 	= vec3.create([0, 0, 0]);
 60 
 61   this.visible   = true;
 62   this.mode = vxl.def.actor.mode.SOLID;
 63   this.opacity = 1.0;
 64   this.colors = model?(model.colors!=null?model.colors:null):null;	//it will create colors for this actor once a lookup table had been set up
 65   this.clones  = 0;
 66   
 67   this._renderers = [];
 68   this._gl_buffers = [];
 69 
 70   
 71   if (model){
 72   	this.model 	 = model;
 73   	this.name 	 = model.name;
 74   	this.diffuse = model.diffuse;
 75   	this.bb 	 = model.bb.slice(0);
 76   	this.mode    = model.mode==undefined?vxl.def.actor.mode.SOLID:model.mode;
 77   }
 78   
 79 };

 82 /**
 83  * Sets the position of this actor. 
 84  * 
 85  * It updates the bounding box according to the position of the actor. The position of the actor
 86  * is initially [0,0,0] and it is relative to the model centre. If the model is not centered in 
 87  * the origin, then the actor's position will be relative to the model centre.
 88  * 
 89  * @param {Number, Array, vec3} x it can be the x coordinate, a 3-dimensional Array or a vec3 (glMatrix)
 90  * @param {Number} y if x is a number, then this parameter corresponds to the y-coordinate
 91  * @param {Number} z if x is a number, then this parameter corresponds to the z-coordinate
 92  */
 93 vxlActor.prototype.setPosition = function (x,y,z){
 94     
 95     var bb = this.bb;
 96     
 97     this.position = vxl.util.createVec3(x,y,z); 
 98     
 99     var currentPos = vec3.set(this.position, vec3.create()); 
100     
101     //Now recalculate the bounding box 
102     var shift = vec3.subtract(this.position, currentPos, vec3.create());
103     bb[0] += shift[0];
104     bb[1] += shift[1];
105     bb[2] += shift[2];
106     bb[3] += shift[0];
107     bb[4] += shift[1];
108     bb[5] += shift[2];
109     
110 };

150 /**
151  * If the property exists, then it updates it
152  * @param {String} property 
153  * @param {Object} value 
154  * @param {String} scope indicates if the change is made at the actor level or at the model level
155  * valid values for scope are vxl.def.model and vxl.def.actor
156  * @TODO: if the property is position or scale then call the respective methods from here
157  */
158 vxlActor.prototype.setProperty = function(property, value, scope){
159     
160     if (property == 'position') {
161     	this.setPosition(value);
162     	return;
163     }
164     
165     if (property == 'scale')    {
166     	this.setScale(value);
167     	return;
168     }
169     
170     if (scope == vxl.def.actor || scope == undefined || scope == null){
171 		this[property] = value;
172 		vxl.go.console('Actor: The actor '+this.name+' has been updated. ['+property+' = '+value+']');
173 	}
174 	else if(scope == vxl.def.model){
175 		this.model[property] = value;
176 	}
177 	else{
178 		throw('vxlActor.setProperty. Scope:' + scope +' is not valid');
179 	}
180 	
181 };

183 /**
184  * Returns an actor property if that property exists in the actor. Otherwise it will search 
185  * in the model. This method is used by the renderer. There are some cases where actors have local changes
186  * that are not reflected in the model. In these cases the renderer should pick the actor property
187  * over the model property
188  * @param{String} property the property name
189  * @returns{Object} the property or undefined if the property is not found 
190  */
191 vxlActor.prototype.getProperty = function(property){
192 	if (this.hasOwnProperty(property)) {
193 		return this[property];
194 	}
195 	else if (this.model.hasOwnProperty(property)){
196 		return this.model[property];
197 	}
198 	else {
199 		return undefined;
200 	}
201 };

203 /**
204  * Estimates the current position as
205  * the center of the current bounding box. 
206  * This method does not update the internal position of the actor
207  * it only returns an estimate based on the location of its bounding box.
208  */
209 vxlActor.prototype.computePosition = function(){
210 	bb = this.bb;
211 	var cc = this.position;
212 	
213 	cc[0] = (bb[3] + bb[0]) /2;
214 	cc[1] = (bb[4] + bb[1]) /2;
215 	cc[2] = (bb[5] + bb[2]) /2;
216 		
217 	cc[0] = Math.round(cc[0]*1000)/1000;
218 	cc[1] = Math.round(cc[1]*1000)/1000;
219 	cc[2] = Math.round(cc[2]*1000)/1000;
220 	
221 	return vec3.create(cc); 
222 };

239 * @param {Number} max highest value for interpolation

242 	if (this.model.scalars){
243 		var lut = vxl.go.lookupTableManager.get(lutID);

271 /**
272  * Duplicates this actor.
273  * 
274  * Properties copied by REFERENCE:
275  * model,
276  * buffers,
277  * renderers
278  * 
279  * Everything else is copied by VALUE.
280  * 
281  * This method is fundamental to replicate objects in the scene, without having to duplicate
282  * the shared model. A cloned actor however can have different position, colors, properties, etc.
283  * 
284  * If a cloned actor modifies his internal model, any other actor that shares the model will be
285  * affected. 
286  * 
287  * The returned actor is not added to the scene automatically. It is up the
288  * programmer to determine the scene the cloned actor needs to be added to if any.
289  * 
290  * @see vxlModel
291  * @returns {vxlActor} an actor 
292  */
293 vxlActor.prototype.clone = function(){
294     this.clones++;
295 	
296 	var duplicate = new vxlActor(this.model);
297 	duplicate['program']   = this['program'];
298 	vec3.set(this.scale,    duplicate.scale);
299 	vec3.set(this.position, duplicate.position);
300 	
301 	
302 	//Now to save us some memory, let's SHARE the WebGL buffers that the current actor has already allocated'
303 	//duplicate.renderers = this.renderers;
304 	//duplicate.buffers   = this.buffers;
305 	//duplicate.model 	= this.model;
306 	duplicate.name     += '-'+this.clones; 
307 	return duplicate;
308 };

311 /**
312 * @private
313 * It gives the strategy the opportunity to allocate memory for this actor. For instance
314 * WebGL buffers.
315 * 
316 * This method is private and it is only supposed to be called by other objects inside
317 * voxelent. Do not invoke directly.
318 */
319 vxlActor.prototype._allocate = function(renderer){
320 	if (this._renderers.indexOf(renderer)!=-1){ //if this renderer has been allocated then ignore
321    		return;
322    }
323    
324    var buffers = renderer._allocateActor(this);
325    
326    this._renderers.push(renderer);
327    this._gl_buffers.push(buffers);
328 
329 };

332 * @private

334 *

336 * voxelent. Do not invoke directly. 
337 */
338 vxlActor.prototype._deallocate = function(renderer){

342 /**
343 * @private

345 * 
346 * This method is private and it is only supposed to be called by other objects inside

348 */
349 vxlActor.prototype._render = function(renderer){
350 	
351 	if (!this.visible){ 
352 		return;
353 	}
354 	renderer._renderActor(this);
