165  /**
166   * Loads 3D models, textures and other models to a Voxelent's scene.
167   * 
168   * This method is very flexible. It can load one or multiple models depending on the type of the 
169   * first parameter. If it is a String, it will try to find the file with that name in Voxelent's data folder
170   * (voxdata by default). Otherwise, if  the parameter 'arguments' is an Array, load will iterate
171   * through it and will try to load every element of this list. Every element being the file name.
172   * 
173   * Nucleo supports three different loading modes which are defined in 
174   * vxl.def.model.loadingMode:
175   * 
176   * LIVE: As each asset is loaded it is added immediately into the scene by creating the corresponding actor
177   * 
178   * LATER: All the models are loaded first THEN the actors are created. 
179   * This is useful when you want to display a full scene instead of showing incremental updates.
180   * 
181   * DETACHED: The models are loaded into the vxlModelManager object but actors are never created.
182   * This allows background loading.
183   * 
184   * @param {String|Array} arguments the name of the asset or the list of models (each element being the file name).
185   * @param {String} path the path that will be concatenated to the list of files (optional).
186   * @param {vxl.def.model.loadingMode} mode the loading mode
187   * @param {vxlScene} scene the scene in case we do not want to load these models in the current one
188   * 
189   * @see {vxl#def#asset#loadingMode}
190   * @see {vxlAssetManager}
191   * @see {vxlScene#setLoadingMode}
192   * 
193   */
194  load :  function(arguments,path,mode,scene){
195  	
196  	var scene = scene==null?vxl.c.scene:scene;
197  	var models = [];
198  	
199  	var p = vxl.util.getPath(path);
200  	
201  	if (typeof(arguments) == 'string' || arguments instanceof String){
202  		models.push(p  + arguments);
203  	}
204  	else if (arguments instanceof Array){
205  		for(var i=0; i<arguments.length;i++){
206 			models.push(p + arguments[i]);
207 		}
208  	}
209  	if (mode != undefined && mode != null){
210 		scene.setLoadingMode(mode);
211 	}
212 	
213 	vxl.go.modelManager.loadList(models, scene);
214  },
