 20 /**
 21  * @private
 22  * creates a new vxlDashEngine
 23  * This engine minimizes the number of drawArray and drawElements calls by
 24  * combining multiple actors in the same VBO.
 25  * 
 26  * vxlDashEngine has replaced the old vxlBakeEngine. 
 27  * vxlDashEnginereduces the number of drawArray, drawElement WebGL calls
 28  * by grouping together similar actors, combining the respective arrays. This is a complex engine
 29  * and there is lot to do (for instance there is no support for big data or picking yet.)
 30  * The new class vxlSilo simplifies the code of vxlDashEngine. vxlSilo*
 31  * grups actors that have the same visualization mode and delivers the javascript arrays and metadata
 32  * that will be passed as attributes and uniforms (respectively) to the ESSL program.
 33  * Pending in vxlDashEngine
 34  * ----------------------------
 35  * 
 36  * SOLID, WIREFRAME, POINTS, LINES, BOUNDING_BOX, BB_AND_SOLID, WIRED_AND_SOLID, FLAT.
 37  * TEXTURED actors: this is a bit more complicated (combining textures, grouping actors by texture used?)
 38  * FLAT actors: related to picking too and to flat mesh representation.
 39  * Picking
 40  *  Multilight support (also pending in vxlRenderEngine)
 41  * 
 42  * 
 43  * 
 44  * 
 45  * 
 46  * @param {Object} renderer
 47  */
 48 function vxlDashEngine(renderer){
 49     vxlEngine.call(this);
 50     this.silos ={};
 51     this._createSilos();
 52     this._gl_buffers = {};
 53     this._createGLBuffers();
 54     this.essl = vxl.util.extend(
 55         vxl.def.essl,{
 56         ACTOR_MATRIX: "aActorMatrix"      
 57     });
 58 };

 61 vxlRenderEngine.prototype.configure = function(){
 62     vxlEngine.prototype.configure.call(this);
 63     var gl = this.gl;
 64     gl.clearDepth(1.0);
 65     gl.disable(gl.CULL_FACE);
 66 };

 68 /**
 69  * @private
 70  */
 71 vxlDashEngine.prototype._createSilos = function(){
 72    for (key in vxl.def.actor.mode){
 73         this.silos[key] = {};
 74         this.silos[key] = new vxlSilo(key);
 75     }
 76 };
 77 
 78 /**
 79  * @private
 80  */
 81 vxlDashEngine.prototype._createGLBuffers = function(){
 82     var gl = this.renderer.gl;
 83     for (key in vxl.def.actor.mode){
 84         this._gl_buffers[key] = {};
 85         this._gl_buffers[key].mixin = gl.createBuffer();
 86         this._gl_buffers[key].index = gl.createBuffer();
 87         this._gl_buffers[key].matrix = gl.createBuffer();
 88     }
 89 };
 90 
 91 /**
 92  * Allocates the scene passed as parameter. This prepares the engine to do the render call
 93  * @param {Object} scene the scene to be rendered
 94  */
 95 vxlDashEngine.prototype.allocate = function(scene){
 96     var actors = scene._actors;
 97     
 98     for (key in this.silos){
 99         this.silos[key].populate(actors);
100         this.silos[key].process();
101     }
102     
103 };
104 
105 /**
106  * Performs the rendering. Unlike the default render
107  * @param {Object} scene the scene to render
108  */
109 vxlDashEngine.prototype.render = function(scene){
110     var r       = this.renderer;
111     var trx     = r.transforms
112     var pm      = r.pm;
113     var gl      = r.gl;
114     var essl    = this.essl;
115     
116     //if there is nothing to render yet then quit
117     if (scene._actors.length == 0) return;
118     
119     //set uniforms
120     trx.update();
121     pm.setUniform(essl.PERSPECTIVE_MATRIX, trx.pMatrix);
122     pm.setUniform(essl.MODEL_VIEW_MATRIX,  trx.mvMatrix);
123     pm.setUniform(essl.NORMAL_MATRIX,      trx.nMatrix);
124     pm.setUniform(essl.MVP_MATRIX,         trx.mvpMatrix);
125     
126     
127     //Renders the actors that have a SOLID visualization mode
128     this._renderSolid();
129     
130 };

140 vxlDashEngine.prototype.enableMatrixAttribute = function(name){
141     var r       = this.renderer;
142     var gl  = r.gl;
143     var prg = r.pm._essl;
144     var m       = vxl.def.actor.mode;
145     var silo    = this.silos[m.SOLID];
146     var data = silo.data;
147     
148     var loc  = gl.getAttribLocation(prg, name);
149     
150     //enable the attribute
151     gl.enableVertexAttribArray(loc);
152     gl.enableVertexAttribArray(loc + 1);
153     gl.enableVertexAttribArray(loc + 2);
154     gl.enableVertexAttribArray(loc + 3);
155     //bind buffer 
156     var matrix0_buffer = gl.createBuffer();
157     var matrix1_buffer = gl.createBuffer();
158     var matrix2_buffer = gl.createBuffer();
159     var matrix3_buffer = gl.createBuffer();
160     
161     gl.bindBuffer(gl.ARRAY_BUFFER, matrix0_buffer);
162     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.matrix[0]), gl.STATIC_DRAW);
163     gl.vertexAttribPointer(loc,4, gl.FLOAT, false, 0, 0);
164     
165     gl.bindBuffer(gl.ARRAY_BUFFER, matrix1_buffer);
166     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.matrix[1]), gl.STATIC_DRAW);
167     gl.vertexAttribPointer(loc+1,4, gl.FLOAT, false, 0, 0);
168     
169     gl.bindBuffer(gl.ARRAY_BUFFER, matrix2_buffer);
170     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.matrix[2]), gl.STATIC_DRAW);
171     gl.vertexAttribPointer(loc+2,4, gl.FLOAT, false, 0, 0);
172     
173     gl.bindBuffer(gl.ARRAY_BUFFER, matrix3_buffer);
174     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.matrix[3]), gl.STATIC_DRAW);
175     gl.vertexAttribPointer(loc+3,4, gl.FLOAT, false, 0, 0);
176     
177     
178     
179 }

182 vxlDashEngine.prototype._renderSolid = function(){
183      var m       = vxl.def.actor.mode;
184      var silo    = this.silos[m.SOLID];
185      var data    = silo.data;
186      var r       = this.renderer;
187      var trx     = r.transforms
188      var pm      = r.pm;
189      var gl      = r.gl;
190      var essl    = this.essl;
191      
192      
193      
194      pm.setUniform("uUseShading",true);
195      
196      pm.enableAttribute(essl.VERTEX_ATTRIBUTE);
197      pm.enableAttribute(essl.NORMAL_ATTRIBUTE);
198      pm.enableAttribute(essl.COLOR_ATTRIBUTE);
199      
200      this.enableMatrixAttribute(essl.ACTOR_MATRIX);
201      
202      var dataVBO = this._gl_buffers[m.SOLID].mixin;
203      gl.bindBuffer(gl.ARRAY_BUFFER, dataVBO);
204      gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(data.mixin), gl.STATIC_DRAW);   
205      pm.setAttributePointer(essl.VERTEX_ATTRIBUTE, 3, gl.FLOAT, false, 36, 0);
206      pm.setAttributePointer(essl.NORMAL_ATTRIBUTE, 3, gl.FLOAT, false, 36, 12);
207      pm.setAttributePointer(essl.COLOR_ATTRIBUTE, 3, gl.FLOAT, false, 36, 24);
208      
209      
210      
211      
212      var indexVBO = this._gl_buffers[m.SOLID].index;
213      gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, indexVBO);
214      gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(data.index), gl.STATIC_DRAW);
215      gl.drawElements(gl.TRIANGLES, data.index.length, gl.UNSIGNED_SHORT,0);
216 };
