 17 /**
 18  * This is a class that interfaces between the current camera and the rendering engine.
 19  * It contains the matrices used by the current camera and that will be used for 
 20  * performing rendering calculations.
 21  * 
 22  * This class also keeps track of the push-pop operations on the model-view matrix stack.
 23  * This is required to combine local and global transformations.
 24  * 
 25  * @class Encapsulates the matrices required to perform 3D rendering
 26  * @constructor
 27  * @author Diego Cantor
 28  * @param {vxlView} p_view the view that this object will refer to.
 29  */
 30 function vxlTransforms(p_view){
 31 	
 32 	this._stack = [];
 33 	this.view = p_view;
 34 	
 35 	this.model_view               = mat4.create();   // The Model-View matrix
 36 	this.projection               = mat4.create();   // The projection matrix
 37 	this.camera                   = mat4.create();   // The camera matrix
 38 	this.normal                   = mat4.create();   // The normal matrix
 39 	this.projection_model_view   = mat4.create();	
 40 };
 41 
 42 /**
 43  * Calculates the current model-view transform.
 44  * This reference is updated whenever the active camera changes.
 45  */
 46 vxlTransforms.prototype.calculateModelView = function(){
 47     //Copy is required so we can do push pop operations
 48 	this.model_view = mat4.copy(this.model_view,this.view.cameraman.active.getViewTransform());
 49     
 50 };
 51 
 52 /**
 53  *Calculates the current camera matrix from the current model-view matrix 
 54  */
 55 vxlTransforms.prototype.calculateCamera = function(){
 56     this.camera = mat4.inverse(this.camera, this.model_view);
 57 };
 58 
 59 /**
 60  * Calculates the normal matrix corresponding to the current Model-View matrix
 61  */
 62 vxlTransforms.prototype.calculateNormal = function(){
 63 	this.normal = mat4.clone(this.model_view);
 64     this.normal = mat4.invert(mat4.create(), this.normal);
 65     this.normal = mat4.transpose(this.normal, this.normal);
 66 };
 67 
 68 /**
 69  * Calculates the projection matrix given the current camera.
 70  * The projection may be orthographic or perspective
 71  */
 72 vxlTransforms.prototype.calculateProjection = function(){
 73     var c = this.view.cameraman.active;
 74     c.updatePerspective();
 75     this.projection = c._perspective;  //for now
 76 };
 77 
 78 /**
 79  * Calculates the projection-model-view matrix 
 80  */
 81 vxlTransforms.prototype.calculateProjectionModelView = function(){
 82     mat4.multiply(this.projection_model_view, this.projection, this.model_view);
 83 };
 84 
 85 /**
 86  * Calculate the transforms for the current view.renderer
 87  * 
 88  */
 89 vxlTransforms.prototype.update = function(){
 90     this.calculateModelView();
 91     this.calculateProjection();
 92     this.calculateNormal();
 93     this.calculateProjectionModelView();
 94 };
 95 
 96 /**
 97  * Saves the current Model-View matrix in the stack. This
 98  * operation is called by vxlActor.updateMatrixStack
 99  * @see vxlActor#updateMatrixStack
100  */
101 
102 vxlTransforms.prototype.push = function(){
103 	var memento =  mat4.create();
104 	mat4.copy(memento, this.model_view);
105 	this._stack.push(memento);
106 };
107 
108 /**
109  * Retrieves the last Model-View transformation in the matrix stack.
110  * This operation is called by vxlActor.updateMatrixStack
111  */
112 vxlTransforms.prototype.pop = function(){
113 	if(this._stack.length == 0) return;
114 	
115 	this.model_view  =  this._stack.pop();
116 	
117 	this.calculatePerspective();
118 	this.calculateNormal();
119 	this.calculateProjectionModelView();
120 };
