1 /**
  2  * @class A Renderable is an intermediary object between an actor and the rendering strategy that allows rendering
  3  * very complex models.
  4  * 
  5  * Think of this scenario. You have a model that has more than 65K number of vertices.
  6  * Yes, it is going to happen eventually.
  7  * 
  8  * In WebGL you can't do more than 65K indices per draw call. This is because the type of the WebGL index array is unsigned short. 
  9  * So, a model with more than 65K indices could not be rendered at once.
 10  * 
 11  * In previous Voxelent versions (<0.89.3) it was required that a complex model was broken down in several
 12  * JSON files. This was a way to make sure that each 'part' did not get so big as to be unable to render it.
 13  * This is, each part had an index array of at most 16k elements.
 14  * 
 15  * For example, the VTK to JSON importer (vtk2json) handles this situation and produces parts with index arrays of length = 65K.
 16  * 
 17  * However, since version 0.89.2, it is possible to create flat shading models and also to do cell based 
 18  * picking. This requires generating models where triangles are not shared (so we can do flat shading).
 19  * Now, if triangles are not shared, then most likely, very complex models are going to surpass the 65K 
 20  * limit for their index array. 
 21  * 
 22  * The renderable object encapsulates a complex model and delivers parts that abide by the 65K index rule.
 23  * 
 24  * The renderables are requested internally by vxlRenderEngine whenever a model of BIG_OBJECT or  MESH types
 25  * need to be rendered.
 26  * 
 27  * 
 28  * 
 29  * @constructor A Renderable is an intermediary object between an actor and the rendering engine that allows rendering
 30  * very complex models.
 31  * 
 32  * @param{vxlActor} model the actor to be decomposed into renderable parts
 33   *  
 34  * @author Diego Cantor
 35  */
 36 
 37 function vxlRenderable(actor){
 38     
 39     
 40     if (actor == undefined){
 41         throw('vxlRenderable: the actor can not be undefined');
 42     }
 43    
 44     
 45     this.actor = actor;
 46     this.parts = [];    
 47     this.update(vxl.def.renderable.task.CREATE);
 48 };

 54 /**
 55  *  Updates the renderable based on changes in the underlying model. After updating, the 
 56  *  renderable parts will contain any changes to the geometry, colors or other model attributes
 57  *  @param {Boolean} reslice if true, it will recreate the parts. Otherwise will use the current parts
 58  */
 59 vxlRenderable.prototype.update = function(task){
 60     
 61     if (task == undefined){
 62         throw ("vxlRenderable.update: Please specify a task");
 63     }
 64 
 65   
 66     var model  = this._getModel();
 67     
 68     if (model == undefined) return; //no model to process
 69     
 70     switch(model.type){
 71         case vxl.def.model.type.MESH:     this._processMesh(model,task); break;
 72         case vxl.def.model.type.BIG_DATA: this._processBigData(model,task); break;
 73     }
 74    
 75 };
 76 
 77 /**
 78  * Queries the actor for a model to process
 79  * @private
 80  */
 81 vxlRenderable.prototype._getModel = function(){
 82     var actor = this.actor;
 83     if (actor.mesh && actor.mesh.model){
 84         return actor.mesh.model;
 85     }
 86     else if (actor.model.type == vxl.def.model.type.BIG_DATA){
 87         return actor.model;
 88     }
 89     else return undefined;
 90 };

 92 /**
 93  * This methods creates renderable parts from a mesh. The implementation is straight forward
 94  * given that a mesh does not share triangles. This is what we need to do flat shading and therefore
 95  * to perform cell color based picking.
 96  * 
 97  * 
 98  */
 99 vxlRenderable.prototype._processMesh = function(model,task){
100     
101   
102 
103     var size = vxl.def.model.MAX_NUM_INDICES;
104     var N = Math.floor(model.indices.length / size), R = model.indices.length % size;
105     
106     if (task == vxl.def.renderable.task.CREATE){
107         
108         this.parts = [];
109         
110         for (var i=0; i<=N; i +=1){
111             
112             var part = new vxlModel(model.name+'-renderable-part-'+i);
113             var startIndex = i*size;
114             var endIndex = startIndex + size;
115             var start = i * size * 3;
116             var end   = start + size *3;
117             
118             if (i == N) {
119                 startIndex = N*size;
120                 endIndex = startIndex + R;
121                 start = N * size * 3;
122                 end   = start + R *3;
123                 
124                 if (R==0){
125                     break;
126                 }
127             }
128             
129             part.indices = this._reindex(model.indices.slice(startIndex,endIndex));
130             part.vertices = model.vertices.slice(start, end);
131             
132             
133             if (model.normals && model.normals.length >0){ part.normals  = model.normals.slice(start, end);  }
134             if (model.colors  && model.colors.length >0) { part.colors = model.colors.slice(start, end);     }
135             
136             if (model.pickingColors){
137                 part.pickingColors = model.pickingColors.slice(start, end);
138             }
139             
140             part.update();
141       
142             this.parts.push(part);
143         }
144     }
145     
146     
147     
148     else if (task == vxl.def.renderable.task.UPDATE_COLORS){
149         for (var i=0; i<=N; i +=1){
150             
151             var part = this.parts[i];
152             var start = i * size * 3;
153             var end   = start + size *3;
154             
155             if (i == N) {
156 
157                 start = N * size * 3;
158                 end   = start + R *3;
159                 
160                 if (R==0){
161                     break;
162                 }
163             }
164           
165             if (model.colors  && model.colors.length >0) { part.colors = model.colors.slice(start, end);  }
166 
167         }
168     }
169      
170 };
171 
172 /**
173  * Return indices starting from 0
174  * @param {Object} indices
175  */
176 vxlRenderable.prototype._reindex = function(indices){
177     var min = indices.min();
178     var i = indices.length;
179     while(i--){
180         indices[i] = indices[i] - min;
181     }
182     return indices;
183 };

186 /***
187  * 
188  * 
189  * The idea here is to read parts of size 65K from the model index array. 
190  * Then obtain vertex, normal, and color arrays for each index in the part (using _getVertexData).
191  * 
192  * As each part is populated, the respective index array is generated accordingly.
193  * 
194  */
195 vxlRenderable.prototype._processBigData = function(model,reslice){
196     
197     var global_index = model.indices;
198     var size         = vxl.def.model.MAX_NUM_INDICES;
199     
200     if (model.mode == vxl.def.actor.mode.LINES){
201         size = size - 1; //adjusting to even number of indices for partitioning.
202     }
203    
204     var L            = global_index.length;
205     var max_index    = global_index.max();
206     var material     = this.actor.material;
207     var has_colors   = (material.colors && material.colors.length>0);
208     var has_normals  = (model.normals   && model.normals.length>0);
209     var has_scalars  = (model.scalars   && model.scalars.length>0);
210     var mode         = model.mode;
211     
212     data = {vertices:[], indices:[],mode:mode};
213     if (has_colors)  { data.colors = [];  }
214     if (has_normals) { data.normals = []; }
215     if (has_scalars) { data.scalars = []; }
216     
217     index_map = {};
218     part_number = 1;
219     new_index = 0;
220     progress = 0;
221     
222     for (var i=0; i<L; i +=1){
223         
224         index = global_index[i];
225         
226         if  (index_map[index] == undefined){
227     
228             index_map[index] = new_index;
229             vertex = this._getVertexData(index);
230             data.vertices.push.apply(data.vertices, vertex.coords);
231                 
232             if (has_normals){
233                 data.normals.push.apply(data.normals, vertex.normal);
234             }
235  
236             if (has_colors){
237                 data.colors.push.apply(data.colors, vertex.color);
238             }
239  
240             if (has_scalars){
241                 data.scalars.push(vertex.scalar);
242             }
243             new_index +=1;
244         }
245         data.indices.push(index_map[index]);
246         
247         if ((new_index == size+1) || (i == L-1)){
248             
249             var part = new vxlModel(model.name+'-renderable-part-'+part_number,data);
250             part.update();
251             vxl.go.console('Creating part '+part.name+' ['+part_number+']',true);    
252             this.parts.push(part);
253             
254             new_index    = 0;
255             part_number += 1;
256             part         = {vertices:[], indices:[],mode:mode};
257             index_map    = {};
258             
259             data = {vertices:[], indices:[],mode:mode};
260             if (has_colors)  { data.colors = [];  }
261             if (has_normals) { data.normals = []; }
262             if (has_scalars) { data.scalars = []; }
263         } 
264     }
265         
266 };

268 /**
269  * Useful to divide a big data model into renderable parts. The method _processBigData
270  * will read information from the buffers indicated by the index in order to populate each
271  * renderable part with the correct information 
272  * 
273  * @param {Object} index
274  * @private
275  */
276 vxlRenderable.prototype._getVertexData = function(index){
277     
278     var material   = this.actor.material;
279     var model      = this.actor.model;
280     var vertex = {};
281     
282     vertex.coords   = model.vertices.slice(index*3, index*3+3);
283     
284     if (model.normals)   {  vertex.normal  = model.normals.slice(index*3, index*3+3);  }
285     if (material.colors) {  vertex.color   = material.colors.slice(index*3, index*3+3);}
286     if (model.scalars)   {  vertex.scalar  = model.scalars[index];}
287     
288     return vertex;
289     
290     
291 };
